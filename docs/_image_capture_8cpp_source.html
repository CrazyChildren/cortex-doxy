57 bool ImageCapture::Open(std::vector<std::string>& arguments)
58 {
59 
60  // Consuming the input arguments
61  bool* valid = new bool[arguments.size()];
62 
63  for (size_t i = 0; i < arguments.size(); ++i)
64  {
65  valid[i] = true;
66  }
67 
68  // Some default values
69  std::string input_root = "";
70  fx = -1; fy = -1; cx = -1; cy = -1;
71 
72  std::string separator = std::string(1, boost::filesystem::path::preferred_separator);
73 
74  // First check if there is a root argument (so that videos and input directories could be defined more easily)
75  for (size_t i = 0; i < arguments.size(); ++i)
76  {
77  if (arguments[i].compare("-root") == 0)
78  {
79  input_root = arguments[i + 1] + separator;
80  i++;
81  }
82  if (arguments[i].compare("-inroot") == 0)
83  {
84  input_root = arguments[i + 1] + separator;
85  i++;
86  }
87  }
88 
89  std::string input_directory;
90  std::string bbox_directory;
91 
92  bool directory_found = false;
93  has_bounding_boxes = false;
94 
95  std::vector<std::string> input_image_files;
96 
97  for (size_t i = 0; i < arguments.size(); ++i)
98  {
99  if (arguments[i].compare("-f") == 0)
100  {
101  input_image_files.push_back(input_root + arguments[i + 1]);
102  valid[i] = false;
103  valid[i + 1] = false;
104  i++;
105  }
106  else if (arguments[i].compare("-fdir") == 0)
107  {
108  if (directory_found)
109  {
110  WARN_STREAM("Input directory already found, using the first one:" + input_directory);
111  }
112  else
113  {
114  input_directory = (input_root + arguments[i + 1]);
115  valid[i] = false;
116  valid[i + 1] = false;
117  i++;
118  directory_found = true;
119  }
120  }
121  else if (arguments[i].compare("-bboxdir") == 0)
122  {
123  bbox_directory = (input_root + arguments[i + 1]);
124  valid[i] = false;
125  valid[i + 1] = false;
126  has_bounding_boxes = true;
127  i++;
128  }
129  else if (arguments[i].compare("-fx") == 0)
130  {
131  std::stringstream data(arguments[i + 1]);
132  data >> fx;
133  i++;
134  }
135  else if (arguments[i].compare("-fy") == 0)
136  {
137  std::stringstream data(arguments[i + 1]);
138  data >> fy;
139  i++;
140  }
141  else if (arguments[i].compare("-cx") == 0)
142  {
143  std::stringstream data(arguments[i + 1]);
144  data >> cx;
145  i++;
146  }
147  else if (arguments[i].compare("-cy") == 0)
148  {
149  std::stringstream data(arguments[i + 1]);
150  data >> cy;
151  i++;
152  }
153  }
154 
155  for (int i = (int)arguments.size() - 1; i >= 0; --i)
156  {
157  if (!valid[i])
158  {
159  arguments.erase(arguments.begin() + i);
160  }
161  }
162 
163  // Based on what was read in open the sequence
164  if (!input_image_files.empty())
165  {
166  return OpenImageFiles(input_image_files, fx, fy, cx, cy);
167  }
168  if (!input_directory.empty())
169  {
170  return OpenDirectory(input_directory, bbox_directory, fx, fy, cx, cy);
171  }
172 
173  // If no input found return false and set a flag for it
174  no_input_specified = true;
175 
176  return false;
177 }

218 {
219  INFO_STREAM("Attempting to read from directory: " << directory);
220 
221  // Setup some defaults
222  frame_num = 0;
223  no_input_specified = false;
224 
225  image_files.clear();
226 
227  boost::filesystem::path image_directory(directory);
228  std::vector<boost::filesystem::path> file_in_directory;
229  copy(boost::filesystem::directory_iterator(image_directory), boost::filesystem::directory_iterator(), back_inserter(file_in_directory));
230 
231  // Sort the images in the directory first
232  sort(file_in_directory.begin(), file_in_directory.end());
233 
234  std::vector<std::string> curr_dir_files;
235 
236  for (std::vector<boost::filesystem::path>::const_iterator file_iterator(file_in_directory.begin()); file_iterator != file_in_directory.end(); ++file_iterator)
237  {
238  // Possible image extension .jpg and .png
239  if (file_iterator->extension().string().compare(".jpg") == 0 || file_iterator->extension().string().compare(".jpeg") == 0 || file_iterator->extension().string().compare(".png") == 0 || file_iterator->extension().string().compare(".bmp") == 0)
240  {
241  curr_dir_files.push_back(file_iterator->string());
242 
243  // If bounding box directory is specified, read the bounding boxes from it
244  if (!bbox_directory.empty())
245  {
246  boost::filesystem::path current_file = *file_iterator;
247  boost::filesystem::path bbox_file = bbox_directory / current_file.filename().replace_extension("txt");
248 
249  // If there is a bounding box file push it to the list of bounding boxes
250  if (boost::filesystem::exists(bbox_file))
251  {
252  std::ifstream in_bbox(bbox_file.string().c_str(), std::ios_base::in);
253 
254  std::vector<cv::Rect_<float> > bboxes_image;
255 
256  // Keep reading bounding boxes from a file, stop if empty line or
257  while (!in_bbox.eof())
258  {
259  std::string bbox_string;
260  std::getline(in_bbox, bbox_string);
261 
262  if (bbox_string.empty())
263  continue;
264

266 
267  float min_x, min_y, max_x, max_y;
268 
269  ss >> min_x >> min_y >> max_x >> max_y;
270  bboxes_image.push_back(cv::Rect_<float>(min_x, min_y, max_x - min_x, max_y - min_y));
271  }
272  in_bbox.close();
273 
274  bounding_boxes.push_back(bboxes_image);
275  }
276  else
277  {
278  ERROR_STREAM("Could not find the corresponding bounding box for file:" + file_iterator->string());
279  exit(1);
280  }
281  }
282  }
283  }
284 
285  image_files = curr_dir_files;
286 
287  if (image_files.empty())
288  {
289  std::cout << "No images found in the directory: " << directory << std::endl;
290  return false;
291  }
292

294  if (fx != -1 && fy != -1)
295  {
296  image_focal_length_set = true;

321 void ImageCapture::SetCameraIntrinsics(float fx, float fy, float cx, float cy)
322 {
323  // If optical centers are not defined just use center of image
324  if (cx == -1)
325  {
326  this->cx = this->image_width / 2.0f;
327  this->cy = this->image_height / 2.0f;
328  }
329  else
330  {
331  this->cx = cx;
332  this->cy = cy;
333  }
334  // Use a rough guess-timate of focal length
335  if (fx == -1)
336  {
337  this->fx = 500.0f * (this->image_width / 640.0f);
338  this->fy = 500.0f * (this->image_height / 480.0f);
339 
340  this->fx = (this->fx + this->fy) / 2.0f;
341  this->fy = this->fx;
342  }
343  else
344  {
345  this->fx = fx;
346  this->fy = fy;
347  }
348 }

373  float _fx = -1;
374  float _fy = -1;

376  if (image_focal_length_set)

378  _fx = fx;
