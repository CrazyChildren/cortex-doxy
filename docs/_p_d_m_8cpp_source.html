58 // Orthonormalising the 3x3 rotation matrix
59 void PDM::Orthonormalise(cv::Matx33f &R)
60 {
61 
62  cv::SVD svd(R, cv::SVD::MODIFY_A);
63 
64  // get the orthogonal matrix from the initial rotation matrix
65  cv::Mat_ <float> X = svd.u*svd.vt;
66 
67  // This makes sure that the handedness is preserved and no reflection happened
68  // by making sure the determinant is 1 and not -1
69  cv::Mat_<float> W = cv::Mat_<float>::eye(3,3);
70  float d = determinant(X);
71  W(2,2) = determinant(X);
72  cv::Mat Rt = svd.u*W*svd.vt;
73 
74  Rt.copyTo(R);
75 
76 }

158 // Get the 2D shape (in image space) from global and local parameters
159 void PDM::CalcShape2D(cv::Mat_<float>& out_shape, const cv::Mat_<float>& params_local, const cv::Vec6f& params_global) const
160 {
161 
162  int n = this->NumberOfPoints();
163 
164  float s = params_global[0]; // scaling factor
165  float tx = params_global[4]; // x offset
166  float ty = params_global[5]; // y offset
167 
168  // get the rotation matrix from the euler angles
169  cv::Vec3f euler(params_global[1], params_global[2], params_global[3]);
170  cv::Matx33f currRot = Utilities::Euler2RotationMatrix(euler);
171 
172  // get the 3D shape of the object
173  cv::Mat_<float> Shape_3D;
174  this->CalcShape3D(Shape_3D, params_local);
175 
176  // create the 2D shape matrix (if it has not been defined yet)
177  if((out_shape.rows != 2 * mean_shape.rows / 3) || (out_shape.cols != 1))
178  {
179  out_shape.create(2*n,1);
180  }
181  // for every vertex
182  for(int i = 0; i < n; i++)
183  {
184  // Transform this using the weak-perspective mapping to 2D from 3D
185  out_shape.at<float>(i ,0) = s * ( currRot(0,0) * Shape_3D.at<float>(i, 0) + currRot(0,1) * Shape_3D.at<float>(i+n ,0) + currRot(0,2) * Shape_3D.at<float>(i+n*2,0) ) + tx;
186  out_shape.at<float>(i+n,0) = s * ( currRot(1,0) * Shape_3D.at<float>(i, 0) + currRot(1,1) * Shape_3D.at<float>(i+n ,0) + currRot(1,2) * Shape_3D.at<float>(i+n*2,0) ) + ty;
187  }
188 }

191 // provided the bounding box of a face and the local parameters (with optional rotation), generates the global parameters that can generate the face with the provided bounding box
192 // This all assumes that the bounding box describes face from left outline to right outline of the face and chin to eyebrows
193 void PDM::CalcParams(cv::Vec6f& out_params_global, const cv::Rect_<float>& bounding_box, const cv::Mat_<float>& params_local, const cv::Vec3f rotation)
194 {
195 
196  // get the shape instance based on local params
197  cv::Mat_<float> current_shape(mean_shape.size());
198 
199  CalcShape3D(current_shape, params_local);
200 
201  // rotate the shape
202  cv::Matx33f rotation_matrix = Utilities::Euler2RotationMatrix(rotation);
203 
204  cv::Mat_<float> reshaped = current_shape.reshape(1, 3);
205 
206  cv::Mat rotated_shape = (cv::Mat(rotation_matrix) * reshaped);
207 
208  // Get the width of expected shape
209  double min_x;
210  double max_x;
211  cv::minMaxLoc(rotated_shape.row(0), &min_x, &max_x);
212 
213  double min_y;
214  double max_y;
215  cv::minMaxLoc(rotated_shape.row(1), &min_y, &max_y);
216 
217  float width = (float) abs(min_x - max_x);
218  float height = (float)abs(min_y - max_y);
219 
220  float scaling = ((bounding_box.width / width) + (bounding_box.height / height)) / 2.0f;
221 
222  // The estimate of face center also needs some correction
223  float tx = bounding_box.x + bounding_box.width / 2;
224  float ty = bounding_box.y + bounding_box.height / 2;
225 
226  // Correct it so that the bounding box is just around the minimum and maximum point in the initialised face
227  tx = tx - scaling * (min_x + max_x)/2.0f;
228  ty = ty - scaling * (min_y + max_y)/2.0f;
229 
230  out_params_global = cv::Vec6f(scaling, rotation[0], rotation[1], rotation[2], tx, ty);
231 }

254 // Calculate the PDM's Jacobian over rigid parameters (rotation, translation and scaling), the additional input W represents trust for each of the landmarks and is part of Non-Uniform RLMS
255 void PDM::ComputeRigidJacobian(const cv::Mat_<float>& p_local, const cv::Vec6f& params_global, cv::Mat_<float> &Jacob, const cv::Mat_<float> W, cv::Mat_<float> &Jacob_t_w)
256 {
257 
258  // number of verts
259  int n = this->NumberOfPoints();
260 
261  Jacob.create(n * 2, 6);
262 
263  float X,Y,Z;
264 
265  float s = params_global[0];
266 
267  cv::Mat_<float> shape_3D;
268  this->CalcShape3D(shape_3D, p_local);
269 
270  // Get the rotation matrix
271  cv::Vec3f euler(params_global[1], params_global[2], params_global[3]);
272  cv::Matx33f currRot = Utilities::Euler2RotationMatrix(euler);
273 
274  float r11 = currRot(0,0);
275  float r12 = currRot(0,1);
276  float r13 = currRot(0,2);
277  float r21 = currRot(1,0);
278  float r22 = currRot(1,1);
279  float r23 = currRot(1,2);
280  float r31 = currRot(2,0);
281  float r32 = currRot(2,1);
282  float r33 = currRot(2,2);
283 
284  cv::MatIterator_<float> Jx = Jacob.begin();
285  cv::MatIterator_<float> Jy = Jx + n * 6;
286 
287  for(int i = 0; i < n; i++)
288  {
289 
290  X = shape_3D.at<float>(i, 0);
291  Y = shape_3D.at<float>(i + n, 0);
292  Z = shape_3D.at<float>(i + n * 2, 0);
293 
294  // The rigid jacobian from the axis angle rotation matrix approximation using small angle assumption (R * R')
295  // where R' = [1, -wz, wy
296  // wz, 1, -wx
297  // -wy, wx, 1]
298  // And this is derived using the small angle assumption on the axis angle rotation matrix parametrisation
299 
300  // scaling term
301  *Jx++ = (X * r11 + Y * r12 + Z * r13);
302  *Jy++ = (X * r21 + Y * r22 + Z * r23);
303 
304  // rotation terms
305  *Jx++ = (s * (Y * r13 - Z * r12) );
306  *Jy++ = (s * (Y * r23 - Z * r22) );
307  *Jx++ = (-s * (X * r13 - Z * r11));
308  *Jy++ = (-s * (X * r23 - Z * r21));
309  *Jx++ = (s * (X * r12 - Y * r11) );
310  *Jy++ = (s * (X * r22 - Y * r21) );
311 
312  // translation terms
313  *Jx++ = 1.0f;
314  *Jy++ = 0.0f;
315  *Jx++ = 0.0f;
316  *Jy++ = 1.0f;
317 
318  }
319 
320  cv::Mat Jacob_w = cv::Mat::zeros(Jacob.rows, Jacob.cols, Jacob.type());
321 
322  Jx = Jacob.begin();
323  Jy = Jx + n*6;
324 
325  cv::MatIterator_<float> Jx_w = Jacob_w.begin<float>();
326  cv::MatIterator_<float> Jy_w = Jx_w + n*6;
327 
328  // Iterate over all Jacobian values and multiply them by the weight in diagonal of W
329  for(int i = 0; i < n; i++)
330  {
331  float w_x = W.at<float>(i, i);
332  float w_y = W.at<float>(i+n, i+n);
333 
334  for(int j = 0; j < Jacob.cols; ++j)
335  {
336  *Jx_w++ = *Jx++ * w_x;
337  *Jy_w++ = *Jy++ * w_y;
338  }
339  }
340 
341  Jacob_t_w = Jacob_w.t();
342 }

345 // Calculate the PDM's Jacobian over all parameters (rigid and non-rigid), the additional input W represents trust for each of the landmarks and is part of Non-Uniform RLMS
346 void PDM::ComputeJacobian(const cv::Mat_<float>& params_local, const cv::Vec6f& params_global, cv::Mat_<float> &Jacobian, const cv::Mat_<float> W, cv::Mat_<float> &Jacob_t_w)
347 {
348 
349  // number of vertices
350  int n = this->NumberOfPoints();
351 
352  // number of non-rigid parameters
353  int m = this->NumberOfModes();
354 
355  Jacobian.create(n * 2, 6 + m);
356 
357  float X,Y,Z;
358 
359  float s = params_global[0];
360 
361  cv::Mat_<float> shape_3D;
362  this->CalcShape3D(shape_3D, params_local);
363 
364  cv::Vec3f euler(params_global[1], params_global[2], params_global[3]);
365  cv::Matx33f currRot = Utilities::Euler2RotationMatrix(euler);
366 
367  float r11 = currRot(0, 0);
368  float r12 = currRot(0, 1);
369  float r13 = currRot(0, 2);
370  float r21 = currRot(1, 0);
371  float r22 = currRot(1, 1);
372  float r23 = currRot(1, 2);
373  float r31 = currRot(2, 0);
374  float r32 = currRot(2, 1);
375  float r33 = currRot(2, 2);
376 
377  cv::MatIterator_<float> Jx = Jacobian.begin();
378  cv::MatIterator_<float> Jy = Jx + n * (6 + m);
379  cv::MatConstIterator_<float> Vx = this->princ_comp.begin();
380  cv::MatConstIterator_<float> Vy = Vx + n*m;
381  cv::MatConstIterator_<float> Vz = Vy + n*m;
382 
383  for(int i = 0; i < n; i++)
384  {
385 
386  X = shape_3D.at<float>(i, 0);
387  Y = shape_3D.at<float>(i + n, 0);
388  Z = shape_3D.at<float>(i + n * 2, 0);
389 
390  // The rigid jacobian from the axis angle rotation matrix approximation using small angle assumption (R * R')
391  // where R' = [1, -wz, wy
392  // wz, 1, -wx
393  // -wy, wx, 1]
394  // And this is derived using the small angle assumption on the axis angle rotation matrix parametrisation
395 
396  // scaling term
397  *Jx++ = (X * r11 + Y * r12 + Z * r13);
398  *Jy++ = (X * r21 + Y * r22 + Z * r23);
399 
400  // rotation terms
401  *Jx++ = (s * (Y * r13 - Z * r12) );
402  *Jy++ = (s * (Y * r23 - Z * r22) );
403  *Jx++ = (-s * (X * r13 - Z * r11));
404  *Jy++ = (-s * (X * r23 - Z * r21));
405  *Jx++ = (s * (X * r12 - Y * r11) );
406  *Jy++ = (s * (X * r22 - Y * r21) );
407 
408  // translation terms
409  *Jx++ = 1.0f;
410  *Jy++ = 0.0f;
411  *Jx++ = 0.0f;
412  *Jy++ = 1.0f;
413 
414  for(int j = 0; j < m; j++,++Vx,++Vy,++Vz)
415  {
416  // How much the change of the non-rigid parameters (when object is rotated) affect 2D motion
417  *Jx++ = ( s*(r11*(*Vx) + r12*(*Vy) + r13*(*Vz)) );
418  *Jy++ = ( s*(r21*(*Vx) + r22*(*Vy) + r23*(*Vz)) );
419  }
420  }
421 
422  // Adding the weights here
423  if(cv::trace(W)[0] != W.rows)
424  {
425  cv::Mat Jacob_w = Jacobian.clone();
426  Jx = Jacobian.begin();
427  Jy = Jx + n*(6+m);
428 
429  cv::MatIterator_<float> Jx_w = Jacob_w.begin<float>();
430  cv::MatIterator_<float> Jy_w = Jx_w + n*(6+m);
431 
432  // Iterate over all Jacobian values and multiply them by the weight in diagonal of W
433  for(int i = 0; i < n; i++)
434  {
435  float w_x = W.at<float>(i, i);
436  float w_y = W.at<float>(i+n, i+n);
437 
438  for(int j = 0; j < Jacobian.cols; ++j)
439  {
440  *Jx_w++ = *Jx++ * w_x;
441  *Jy_w++ = *Jy++ * w_y;
442  }
443  }
444  Jacob_t_w = Jacob_w.t();
445  }
446  else
447  {
448  Jacob_t_w = Jacobian.t();
449  }
450 }

453 // Updating the parameters (more details in my thesis)
454 void PDM::UpdateModelParameters(const cv::Mat_<float>& delta_p, cv::Mat_<float>& params_local, cv::Vec6f& params_global)
455 {
456 
457  // The scaling and translation parameters can be just added
458  params_global[0] += delta_p.at<float>(0,0);
459  params_global[4] += delta_p.at<float>(4,0);
460  params_global[5] += delta_p.at<float>(5,0);
461 
462  // get the original rotation matrix
463  cv::Vec3f eulerGlobal(params_global[1], params_global[2], params_global[3]);
464 
465  cv::Matx33f R1 = Utilities::Euler2RotationMatrix(eulerGlobal);
466 
467  // construct R' = [1, -wz, wy
468  // wz, 1, -wx
469  // -wy, wx, 1]
470  cv::Matx33f R2 = cv::Matx33f::eye();
471 
472  R2(1,2) = -1.0*(R2(2,1) = delta_p.at<float>(1,0));
473  R2(2,0) = -1.0*(R2(0,2) = delta_p.at<float>(2,0));
474  R2(0,1) = -1.0*(R2(1,0) = delta_p.at<float>(3,0));
475 
476  // Make sure it's orthonormal
477  Orthonormalise(R2);
478 
479  // Combine rotations
480  cv::Matx33f R3 = R1 *R2;
481 
482  // Extract euler angle (through axis angle first to make sure it's legal)
483  cv::Vec3f axis_angle = Utilities::RotationMatrix2AxisAngle(R3);
484 
485  cv::Vec3f euler = Utilities::AxisAngle2Euler(axis_angle);
486 
487  // Temporary fix to numerical instability
488  if (isnan(euler[0]) || isnan(euler[1]) || isnan(euler[2]))
489  {
490  euler[0] = 0;
491  euler[1] = 0;
492  euler[2] = 0;
493 
494  }
495 
496  params_global[1] = euler[0];
497  params_global[2] = euler[1];
498  params_global[3] = euler[2];
499 
500  // Local parameter update, just simple addition
501  if(delta_p.rows > 6)
502  {
503  params_local = params_local + delta_p(cv::Rect(0,6,1, this->NumberOfModes()));
504  }
505 
506 }

508 void PDM::CalcParams(cv::Vec6f& out_params_global, cv::Mat_<float>& out_params_local, const cv::Mat_<float> & landmark_locations, const cv::Vec3f rotation)
509 {
510 
511  int m = this->NumberOfModes();
512  int n = this->NumberOfPoints();
513 
514  cv::Mat_<int> visi_ind_2D(n * 2, 1, 1);
515  cv::Mat_<int> visi_ind_3D(3 * n , 1, 1);
516 
517  int visi_count = n;
518 
519  for(int i = 0; i < n; ++i)
520  {
521  // If the landmark is invisible indicate this
522  if(landmark_locations.at<float>(i) == 0)
523  {
524  visi_ind_2D.at<int>(i) = 0;
525  visi_ind_2D.at<int>(i+n) = 0;
526  visi_ind_3D.at<int>(i) = 0;
527  visi_ind_3D.at<int>(i+n) = 0;
528  visi_ind_3D.at<int>(i+2*n) = 0;
529 
530  visi_count--;
531  }
532  }
533 
534  // As not all landmarks might be visible, subsample the Mean and principal component matrices
535  cv::Mat_<float> M(visi_count * 3, mean_shape.cols, 0.0);
536  cv::Mat_<float> V(visi_count * 3, princ_comp.cols, 0.0);
537  visi_count = 0;
538  for (int i = 0; i < n * 3; ++i)
539  {
540  if (visi_ind_3D.at<int>(i) == 1)
541  {
542  this->mean_shape.row(i).copyTo(M.row(visi_count));
543  this->princ_comp.row(i).copyTo(V.row(visi_count));
544  visi_count++;
545  }
546  }
547 
548  cv::Mat_<float> m_old = this->mean_shape.clone();
549  cv::Mat_<float> v_old = this->princ_comp.clone();
550 
551  this->mean_shape = M;
552  this->princ_comp = V;
553 
554  // The new number of points
555  n = M.rows / 3;
556 
557  // Extract the relevant landmark locations
558  cv::Mat_<float> landmark_locs_vis(n*2, 1, 0.0f);
559  int k = 0;
560  for(int i = 0; i < visi_ind_2D.rows; ++i)
561  {
562  if(visi_ind_2D.at<int>(i) == 1)
563  {
564  landmark_locs_vis.at<float>(k) = landmark_locations.at<float>(i);
565  k++;
566  }
567  }
568 
569  // Compute the initial global parameters
570  float min_x, max_x, min_y, max_y;
571  ExtractBoundingBox(landmark_locs_vis, min_x, max_x, min_y, max_y);
572
