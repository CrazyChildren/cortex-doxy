70 cv::Vec6f LandmarkDetector::GetPose(const CLNF& clnf_model, float fx, float fy, float cx, float cy)
71 {
72  if (!clnf_model.detected_landmarks.empty() && clnf_model.params_global[0] != 0)
73  {
74  // This is used as an initial estimate for the iterative PnP algorithm
75  float Z = fx / clnf_model.params_global[0];
76 
77  float X = ((clnf_model.params_global[4] - cx) * (1.0 / fx)) * Z;
78  float Y = ((clnf_model.params_global[5] - cy) * (1.0 / fy)) * Z;
79 
80  // Correction for orientation
81 
82  // 2D points
83  cv::Mat_<float> landmarks_2D = clnf_model.detected_landmarks;
84 
85  landmarks_2D = landmarks_2D.reshape(1, 2).t();
86  //featurePoint=landmarks_2D;
87  // 3D points
88  cv::Mat_<float> landmarks_3D;
89  clnf_model.pdm.CalcShape3D(landmarks_3D, clnf_model.params_local);
90 
91  landmarks_3D = landmarks_3D.reshape(1, 3).t();
92 
93  // Solving the PNP model
94 
95  // The camera matrix
96  cv::Matx33f camera_matrix(fx, 0, cx, 0, fy, cy, 0, 0, 1);
97 
98  cv::Vec3f vec_trans(X, Y, Z);
99  cv::Vec3f vec_rot(clnf_model.params_global[1], clnf_model.params_global[2], clnf_model.params_global[3]);
100 
101  cv::solvePnP(landmarks_3D, landmarks_2D, camera_matrix, cv::Mat(), vec_rot, vec_trans, true);
102 
103  cv::Vec3f euler = Utilities::AxisAngle2Euler(vec_rot);
104 
105  return cv::Vec6f(vec_trans[0], vec_trans[1], vec_trans[2], euler[0], euler[1], euler[2]);
106  }
107  else
108  {
109  return cv::Vec6f(0, 0, 0, 0, 0, 0);
110  }
111 }

113 // Getting a head pose estimate from the currently detected landmarks, with appropriate correction due to perspective projection
114 // This method returns a corrected pose estimate with respect to a point camera (NOTE not the world coordinates), which is useful to find out if the person is looking at a camera
115 // The format returned is [Tx, Ty, Tz, Eul_x, Eul_y, Eul_z]
116 cv::Vec6f LandmarkDetector::GetPoseWRTCamera(const CLNF& clnf_model, float fx, float fy, float cx, float cy)
117 {
118  if (!clnf_model.detected_landmarks.empty() && clnf_model.params_global[0] != 0)
119  {
120 
121  float Z = fx / clnf_model.params_global[0];
122 
123  float X = ((clnf_model.params_global[4] - cx) * (1.0 / fx)) * Z;
124  float Y = ((clnf_model.params_global[5] - cy) * (1.0 / fy)) * Z;
125 
126  // Correction for orientation
127 
128  // 3D points
129  cv::Mat_<float> landmarks_3D;
130  clnf_model.pdm.CalcShape3D(landmarks_3D, clnf_model.params_local);
131 
132  landmarks_3D = landmarks_3D.reshape(1, 3).t();
133 
134  // 2D points
135  cv::Mat_<float> landmarks_2D = clnf_model.detected_landmarks;
136 
137  landmarks_2D = landmarks_2D.reshape(1, 2).t();
138 
139  // Solving the PNP model
140 
141  // The camera matrix
142  cv::Matx33f camera_matrix(fx, 0, cx, 0, fy, cy, 0, 0, 1);
143 
144  cv::Vec3f vec_trans(X, Y, Z);
145  cv::Vec3f vec_rot(clnf_model.params_global[1], clnf_model.params_global[2], clnf_model.params_global[3]);
146 
147  cv::solvePnP(landmarks_3D, landmarks_2D, camera_matrix, cv::Mat(), vec_rot, vec_trans, true);
148 
149  // Here we correct for the camera orientation, for this need to determine the angle the camera makes with the head pose
150  float z_x = cv::sqrt(vec_trans[0] * vec_trans[0] + vec_trans[2] * vec_trans[2]);
151  float eul_x = atan2(vec_trans[1], z_x);
152 
153  float z_y = cv::sqrt(vec_trans[1] * vec_trans[1] + vec_trans[2] * vec_trans[2]);
154  float eul_y = -atan2(vec_trans[0], z_y);
155 
156  cv::Matx33f camera_rotation = Utilities::Euler2RotationMatrix(cv::Vec3f(eul_x, eul_y, 0));
157  cv::Matx33f head_rotation = Utilities::AxisAngle2RotationMatrix(vec_rot);
158 
159  cv::Matx33f corrected_rotation = camera_rotation * head_rotation;
160 
161  cv::Vec3f euler_corrected = Utilities::RotationMatrix2Euler(corrected_rotation);
162 
163  return cv::Vec6f(vec_trans[0], vec_trans[1], vec_trans[2], euler_corrected[0], euler_corrected[1], euler_corrected[2]);
164  }
165  else
166  {
167  return cv::Vec6f(0, 0, 0, 0, 0, 0);
168  }

185 void CorrectGlobalParametersVideo(const cv::Mat_<uchar> &grayscale_image, CLNF& clnf_model, const FaceModelParameters& params)
186 {
187  cv::Rect_<float> init_box;
188  clnf_model.pdm.CalcBoundingBox(init_box, clnf_model.params_global, clnf_model.params_local);
189 
190  cv::Rect roi(init_box.x - init_box.width/2, init_box.y - init_box.height/2, init_box.width * 2, init_box.height * 2);
191  roi = roi & cv::Rect(0, 0, grayscale_image.cols, grayscale_image.rows);
192 
193  int off_x = roi.x;
194  int off_y = roi.y;
195 
196  float scaling = params.face_template_scale / clnf_model.params_global[0];
197  cv::Mat_<uchar> image;
198  if(scaling < 1)
199  {
200  cv::resize(clnf_model.face_template, clnf_model.face_template, cv::Size(), scaling, scaling);
201  cv::resize(grayscale_image(roi), image, cv::Size(), scaling, scaling);
202  }
203  else
204  {
205  scaling = 1;
206  image = grayscale_image(roi).clone();
207  }
208 
209  // Resizing the template
210  cv::Mat corr_out;
211  cv::matchTemplate(image, clnf_model.face_template, corr_out, cv::TM_CCOEFF_NORMED);
212 
213  // Actually matching it
214  //double min, max;
215  int max_loc[2];
216 
217  cv::minMaxIdx(corr_out, NULL, NULL, NULL, max_loc);
218 
219  cv::Rect_<float> out_bbox(max_loc[1]/scaling + off_x, max_loc[0]/scaling + off_y, clnf_model.face_template.rows / scaling, clnf_model.face_template.cols / scaling);
220 
221  float shift_x = out_bbox.x - init_box.x;
222  float shift_y = out_bbox.y - init_box.y;
223 
224  clnf_model.params_global[4] = clnf_model.params_global[4] + shift_x;
225  clnf_model.params_global[5] = clnf_model.params_global[5] + shift_y;

229 bool LandmarkDetector::DetectLandmarksInVideo(const cv::Mat &rgb_image, CLNF& clnf_model, FaceModelParameters& params, cv::Mat& grayscale_image)
230 {
231  // First need to decide if the landmarks should be "detected" or "tracked"
232  // Detected means running face detection and a larger search area, tracked means initialising from previous step
233  // and using a smaller search area
234 
235  if(grayscale_image.empty())
236  {
237  Utilities::ConvertToGrayscale_8bit(rgb_image, grayscale_image);
238  }
239 
240  // Indicating that this is a first detection in video sequence or after restart
241  bool initial_detection = !clnf_model.tracking_initialised;
242 
243  // Only do it if there was a face detection at all
244  if(clnf_model.tracking_initialised)
245  {
246 
247  // The area of interest search size will depend if the previous track was successful
248  if(!clnf_model.detection_success)
249  {
250  params.window_sizes_current = params.window_sizes_init;
251  }
252  else
253  {
254  params.window_sizes_current = params.window_sizes_small;
255  }
256 
257  // Before the expensive landmark detection step apply a quick template tracking approach
258  if(params.use_face_template && !clnf_model.face_template.empty() && clnf_model.detection_success)
259  {
260  CorrectGlobalParametersVideo(grayscale_image, clnf_model, params);
261  }
262 
263  bool track_success = clnf_model.DetectLandmarks(grayscale_image, params);
264 
265  if(!track_success)
266  {
267  // Make a record that tracking failed
268  clnf_model.failures_in_a_row++;
269  }
270  else
271  {
272  // indicate that tracking is a success
273  clnf_model.failures_in_a_row = -1;
274 
275  if(params.use_face_template)
276  {
277  UpdateTemplate(grayscale_image, clnf_model);
278  }
279  }
280  }
281 
282  // This is used for both detection (if it the tracking has not been initialised yet) or if the tracking failed (however we do this every n frames, for speed)
283  // This also has the effect of an attempt to reinitialise just after the tracking has failed, which is useful during large motions
284  if((!clnf_model.tracking_initialised && (clnf_model.failures_in_a_row + 1) % (params.reinit_video_every * 6) == 0)
285  || (clnf_model.tracking_initialised && !clnf_model.detection_success && params.reinit_video_every > 0 && clnf_model.failures_in_a_row % params.reinit_video_every == 0))
286  {
287 
288  cv::Rect_<float> bounding_box;
289 
290  // If the face detector has not been initialised and we're using it, then read it in
291  if(clnf_model.face_detector_HAAR.empty() && params.curr_face_detector == params.HAAR_DETECTOR)
292  {
293  clnf_model.face_detector_HAAR.load(params.haar_face_detector_location);
294  clnf_model.haar_face_detector_location = params.haar_face_detector_location;
295  }
296  if (clnf_model.face_detector_MTCNN.empty() && params.curr_face_detector == params.MTCNN_DETECTOR)
297  {
298  clnf_model.face_detector_MTCNN.Read(params.mtcnn_face_detector_location);
299  clnf_model.mtcnn_face_detector_location = params.mtcnn_face_detector_location;
300 
301  // If the model is still empty default to HOG
302  if (clnf_model.face_detector_MTCNN.empty())
303  {
304  cout << "INFO: defaulting to HOG-SVM face detector" << endl;
305  params.curr_face_detector = LandmarkDetector::FaceModelParameters::HOG_SVM_DETECTOR;
306  }
307 
308  }
309 
310  cv::Point preference_det(-1, -1);
311  if(clnf_model.preference_det.x != -1 && clnf_model.preference_det.y != -1)
312  {
313  preference_det.x = clnf_model.preference_det.x * grayscale_image.cols;
314  preference_det.y = clnf_model.preference_det.y * grayscale_image.rows;
315  clnf_model.preference_det = cv::Point(-1, -1);
316  }
317 
318  bool face_detection_success;
319  if(params.curr_face_detector == FaceModelParameters::HOG_SVM_DETECTOR)
320  {
321  float confidence;
322  face_detection_success = LandmarkDetector::DetectSingleFaceHOG(bounding_box, grayscale_image, clnf_model.face_detector_HOG, confidence, preference_det);
323  }
324  else if(params.curr_face_detector == FaceModelParameters::HAAR_DETECTOR)
325  {
326  face_detection_success = LandmarkDetector::DetectSingleFace(bounding_box, grayscale_image, clnf_model.face_detector_HAAR, preference_det);
327  }
328  else if (params.curr_face_detector == FaceModelParameters::MTCNN_DETECTOR)
329  {
330  float confidence;
331  face_detection_success = LandmarkDetector::DetectSingleFaceMTCNN(bounding_box, rgb_image, clnf_model.face_detector_MTCNN, confidence, preference_det);
332  }
333 
334  // Attempt to detect landmarks using the detected face (if unseccessful the detection will be ignored)
335  if(face_detection_success)
336  {
337  // Indicate that tracking has started as a face was detected
338  clnf_model.tracking_initialised = true;
339 
340  // Keep track of old model values so that they can be restored if redetection fails
341  cv::Vec6f params_global_init = clnf_model.params_global;
342  cv::Mat_<float> params_local_init = clnf_model.params_local.clone();
343  float likelihood_init = clnf_model.model_likelihood;
344  cv::Mat_<float> detected_landmarks_init = clnf_model.detected_landmarks.clone();
345  cv::Mat_<float> landmark_likelihoods_init = clnf_model.landmark_likelihoods.clone();
346 
347  // Use the detected bounding box and empty local parameters
348  clnf_model.params_local.setTo(0);
349  clnf_model.pdm.CalcParams(clnf_model.params_global, bounding_box, clnf_model.params_local);
350 
351  // Make sure the search size is large
352  params.window_sizes_current = params.window_sizes_init;
353 
354  // TODO rem (should the multi-hyp version be only for CEN and not CLNF?), otherwise poss too slow, and poss not accurate
355  //bool landmark_detection_success = clnf_model.DetectLandmarks(grayscale_image, params);
356 
357  // Do the actual landmark detection (and keep it only if successful)
358  // Perform multi-hypothesis detection here (as face detector can pick up multiple of them)
359  params.multi_view = true;
360  bool landmark_detection_success = DetectLandmarksInImage(rgb_image, bounding_box, clnf_model, params, grayscale_image);
361  params.multi_view = false;
362 
363 
364  // If landmark reinitialisation unsucessful continue from previous estimates
365  // if it's initial detection however, do not care if it was successful as the validator might be wrong, so continue trackig
366  // regardless
367  if(!initial_detection && !landmark_detection_success)
368  {
369 
370  // Restore previous estimates
371  clnf_model.params_global = params_global_init;
372  clnf_model.params_local = params_local_init.clone();
373  clnf_model.pdm.CalcShape2D(clnf_model.detected_landmarks, clnf_model.params_local, clnf_model.params_global);
374  clnf_model.model_likelihood = likelihood_init;
375  clnf_model.detected_landmarks = detected_landmarks_init.clone();
376  clnf_model.landmark_likelihoods = landmark_likelihoods_init.clone();
377 
378  return false;
379  }
380  else
381  {
382  clnf_model.failures_in_a_row = -1;
383 
384  if(params.use_face_template)
385  {
386  UpdateTemplate(grayscale_image, clnf_model);
387  }
388 
389  return true;
390  }
391  }
392  }
393 
394  // if the model has not been initialised yet class it as a failure
395  if(!clnf_model.tracking_initialised)
396  {
397  clnf_model.failures_in_a_row++;
398  }
399 
400  // un-initialise the tracking
401  if( clnf_model.failures_in_a_row > 100)
402  {
403  clnf_model.tracking_initialised = false;
404  }
405 
406  return clnf_model.detection_success;
407 
408 }

431 bool DetectLandmarksInImageMultiHypBasic(const cv::Mat_<uchar> &grayscale_image, vector<cv::Vec3d> rotation_hypotheses, const cv::Rect_<double> bounding_box, CLNF& clnf_model, FaceModelParameters& params)
432 {
433 
434  // Use the initialisation size for the landmark detection
435  params.window_sizes_current = params.window_sizes_init;
436 
437  // Store the current best estimate
438  float best_likelihood;
439  float best_detection_certainty;
440  cv::Vec6f best_global_parameters;
441  cv::Mat_<float> best_local_parameters;
442  cv::Mat_<float> best_detected_landmarks;
443  cv::Mat_<float> best_landmark_likelihoods;
444  bool best_success;
445 
446  // The hierarchical model parameters
447  vector<float> best_likelihood_h(clnf_model.hierarchical_models.size());
448  vector<cv::Vec6f> best_global_parameters_h(clnf_model.hierarchical_models.size());
449  vector<cv::Mat_<float>> best_local_parameters_h(clnf_model.hierarchical_models.size());
450  vector<cv::Mat_<float>> best_detected_landmarks_h(clnf_model.hierarchical_models.size());
451  vector<cv::Mat_<float>> best_landmark_likelihoods_h(clnf_model.hierarchical_models.size());
452 
453  for (size_t hypothesis = 0; hypothesis < rotation_hypotheses.size(); ++hypothesis)
454  {
455  // Reset the potentially set clnf_model parameters
456  clnf_model.params_local.setTo(0.0);
457 
458  for (size_t part = 0; part < clnf_model.hierarchical_models.size(); ++part)
459  {
460  clnf_model.hierarchical_models[part].params_local.setTo(0.0);
461  }
462 
463  // calculate the local and global parameters from the generated 2D shape (mapping from the 2D to 3D because camera params are unknown)
464  clnf_model.pdm.CalcParams(clnf_model.params_global, bounding_box, clnf_model.params_local, rotation_hypotheses[hypothesis]);
465 
466  bool success = clnf_model.DetectLandmarks(grayscale_image, params);
467 
468  if (hypothesis == 0 || best_likelihood < clnf_model.model_likelihood)
469  {
470  best_likelihood = clnf_model.model_likelihood;
471  best_global_parameters = clnf_model.params_global;
472  best_local_parameters = clnf_model.params_local.clone();
473  best_detected_landmarks = clnf_model.detected_landmarks.clone();
474  best_landmark_likelihoods = clnf_model.landmark_likelihoods.clone();
475  best_detection_certainty = clnf_model.detection_certainty;
476  best_success = success;
477 
478  for (size_t part = 0; part < clnf_model.hierarchical_models.size(); ++part)
479  {
480  best_likelihood_h[part] = clnf_model.hierarchical_models[part].model_likelihood;
481  best_global_parameters_h[part] = clnf_model.hierarchical_models[part].params_global;
482  best_local_parameters_h[part] = clnf_model.hierarchical_models[part].params_local.clone();
483  best_detected_landmarks_h[part] = clnf_model.hierarchical_models[part].detected_landmarks.clone();
484  best_landmark_likelihoods_h[part] = clnf_model.hierarchical_models[part].landmark_likelihoods.clone();
485  }
486  }
487 
488  }
489 
490  // Store the best estimates in the clnf_model
491  clnf_model.model_likelihood = best_likelihood;
492  clnf_model.params_global = best_global_parameters;
493  clnf_model.params_local = best_local_parameters.clone();
494  clnf_model.detected_landmarks = best_detected_landmarks.clone();
495  clnf_model.detection_success = best_success;
496  clnf_model.landmark_likelihoods = best_landmark_likelihoods.clone();
497  clnf_model.detection_certainty = best_detection_certainty;
498 
499  for (size_t part = 0; part < clnf_model.hierarchical_models.size(); ++part)
500  {
501  clnf_model.hierarchical_models[part].params_global = best_global_parameters_h[part];
502  clnf_model.hierarchical_models[part].params_local = best_local_parameters_h[part].clone();
503  clnf_model.hierarchical_models[part].detected_landmarks = best_detected_landmarks_h[part].clone();
504  clnf_model.hierarchical_models[part].landmark_likelihoods = best_landmark_likelihoods_h[part].clone();
505  }
506 
507  return best_success;

513 template <typename T> std::vector<size_t> sort_indexes(const vector<T> &v) {
514 
515  // initialize original index locations
516  vector<size_t> idx(v.size());

526 bool DetectLandmarksInImageMultiHypEarlyTerm(const cv::Mat_<uchar> &grayscale_image, vector<cv::Vec3d> rotation_hypotheses, const cv::Rect_<double> bounding_box, CLNF& clnf_model, FaceModelParameters& params)
527 {
528  FaceModelParameters old_params(params);
529 
530  // Use the initialisation size for the landmark detection
531  params.window_sizes_current = params.window_sizes_init;
532 
533  bool early_term = false;
534 
535  // Setup the parameters accordingly
536  // Only do the first iteration
537  for (size_t i = 1; i < params.window_sizes_current.size(); ++i)
538  {
539  params.window_sizes_current[i] = 0;
540  }
541  params.refine_hierarchical = false;
542  params.validate_detections = false;
543 
544  bool success = false;
545 
546  // Keeping track of converges
547  vector<float> likelihoods;
548  vector<cv::Vec6f> global_parameters;
549  vector<cv::Mat_<float>> local_parameters;
550 
551  for (size_t hypothesis = 0; hypothesis < rotation_hypotheses.size(); ++hypothesis)
552  {
553  // Reset the potentially set clnf_model parameters
554  clnf_model.params_local.setTo(0.0);
555 
556  for (size_t part = 0; part < clnf_model.hierarchical_models.size(); ++part)
557  {
558  clnf_model.hierarchical_models[part].params_local.setTo(0.0);
559  }
560 
561  // calculate the local and global parameters from the generated 2D shape (mapping from the 2D to 3D because camera params are unknown)
562  clnf_model.pdm.CalcParams(clnf_model.params_global, bounding_box, clnf_model.params_local, rotation_hypotheses[hypothesis]);
563 
564  // Perform landmark detection in first scale
565  clnf_model.DetectLandmarks(grayscale_image, params);
566 
567  float lhood = clnf_model.model_likelihood * clnf_model.patch_experts.early_term_weights[clnf_model.view_used] + clnf_model.patch_experts.early_term_biases[clnf_model.view_used];
568 
569  // If likelihood higher than cutoff continue on this model
570  if (lhood > clnf_model.patch_experts.early_term_cutoffs[clnf_model.view_used])
571  {
572  params.refine_hierarchical = old_params.refine_hierarchical;
573  params.window_sizes_current = params.window_sizes_init;
574  params.window_sizes_current[0] = 0;
575  params.validate_detections = old_params.validate_detections;
576  success = clnf_model.DetectLandmarks(grayscale_image, params);
577  early_term = true;
578  break;
579  }
580  else
581  {
582  likelihoods.push_back(lhood);
583  global_parameters.push_back(clnf_model.params_global);
584  local_parameters.push_back(clnf_model.params_local);
585  }
586  }
587 
588 
589  if (!early_term)
590  {
591 
592  // Store the current best estimate
593  float best_likelihood;
594  cv::Vec6f best_global_parameters;
595  cv::Mat_<float> best_local_parameters;
596  cv::Mat_<float> best_detected_landmarks;
597  cv::Mat_<float> best_landmark_likelihoods;
598  bool best_success;
599 
600  // The hierarchical model parameters
601  vector<float> best_likelihood_h(clnf_model.hierarchical_models.size());
602  vector<cv::Vec6f> best_global_parameters_h(clnf_model.hierarchical_models.size());
603  vector<cv::Mat_<float>> best_local_parameters_h(clnf_model.hierarchical_models.size());
604  vector<cv::Mat_<float>> best_detected_landmarks_h(clnf_model.hierarchical_models.size());
605  vector<cv::Mat_<float>> best_landmark_likelihoods_h(clnf_model.hierarchical_models.size());
606 
607  // Sort the likelihoods and pick the best top 3 models
608  vector<size_t> indices = sort_indexes(likelihoods);
609 
610  // Pick 3 best hypotheses and complete them
611  size_t max = indices.size() >= 3 ? 3 : indices.size();
612 
613  params.refine_hierarchical = old_params.refine_hierarchical;
614  params.window_sizes_current = params.window_sizes_init;
615  params.window_sizes_current[0] = 0;
616  params.validate_detections = old_params.validate_detections;
617 
618 
619  for (size_t i = 0; i < max; ++i)
620  {
621  // Reset the potentially set clnf_model parameters
622  clnf_model.params_local = local_parameters[indices[i]];
623  clnf_model.params_global = global_parameters[indices[i]];
624  for (size_t part = 0; part < clnf_model.hierarchical_models.size(); ++part)
625  {
626  clnf_model.hierarchical_models[part].params_local.setTo(0.0);
627  }
628 
629  // Perform landmark detection in first scale
630  success = clnf_model.DetectLandmarks(grayscale_image, params);
631 
632  if (i == 0 || best_likelihood < clnf_model.model_likelihood)
633  {
634  best_likelihood = clnf_model.model_likelihood;
635  best_global_parameters = clnf_model.params_global;
636  best_local_parameters = clnf_model.params_local.clone();
637  best_detected_landmarks = clnf_model.detected_landmarks.clone();
638  best_landmark_likelihoods = clnf_model.landmark_likelihoods.clone();
639  best_success = success;
640 
641  for (size_t part = 0; part < clnf_model.hierarchical_models.size(); ++part)
642  {
643  best_likelihood_h[part] = clnf_model.hierarchical_models[part].model_likelihood;
644  best_global_parameters_h[part] = clnf_model.hierarchical_models[part].params_global;
645  best_local_parameters_h[part] = clnf_model.hierarchical_models[part].params_local.clone();
646  best_detected_landmarks_h[part] = clnf_model.hierarchical_models[part].detected_landmarks.clone();
647  best_landmark_likelihoods_h[part] = clnf_model.hierarchical_models[part].landmark_likelihoods.clone();
648  }
649  }
650 
651  }
652 
653  // Store the best estimates in the clnf_model
654  clnf_model.model_likelihood = best_likelihood;
655  clnf_model.params_global = best_global_parameters;
656  clnf_model.params_local = best_local_parameters.clone();
657  clnf_model.detected_landmarks = best_detected_landmarks.clone();
658  clnf_model.detection_success = best_success;
659  clnf_model.landmark_likelihoods = best_landmark_likelihoods.clone();
660 
661  for (size_t part = 0; part < clnf_model.hierarchical_models.size(); ++part)
662  {
663  clnf_model.hierarchical_models[part].params_global = best_global_parameters_h[part];
664  clnf_model.hierarchical_models[part].params_local = best_local_parameters_h[part].clone();
665  clnf_model.hierarchical_models[part].detected_landmarks = best_detected_landmarks_h[part].clone();
666  clnf_model.hierarchical_models[part].landmark_likelihoods = best_landmark_likelihoods_h[part].clone();
667  }
668 
669  }
670 
671  params = old_params;
672 
673  return success;
674
