337 // Calculate the warping coefficients
338 void PAW::CalcCoeff()
339 {
340  int p = this->NumberOfLandmarks();
341 
342  for (int l = 0; l < this->NumberOfTriangles(); l++)
343  {
344 
345  int i = triangulation.at<int>(l, 0);
346  int j = triangulation.at<int>(l, 1);
347  int k = triangulation.at<int>(l, 2);
348 
349  float c1 = source_landmarks.at<float>(i, 0);
350  float c2 = source_landmarks.at<float>(j, 0) - c1;
351  float c3 = source_landmarks.at<float>(k, 0) - c1;
352  float c4 = source_landmarks.at<float>(i + p, 0);
353  float c5 = source_landmarks.at<float>(j + p, 0) - c4;
354  float c6 = source_landmarks.at<float>(k + p, 0) - c4;
355 
356  // Get a pointer to the coefficient we will be precomputing
357  float *coeff = coefficients.ptr<float>(l);
358 
359  // Extract the relevant alphas and betas
360  float *c_alpha = alpha.ptr<float>(l);
361  float *c_beta = beta.ptr<float>(l);
362 
363  coeff[0] = c1 + c2 * c_alpha[0] + c3 * c_beta[0];
364  coeff[1] = c2 * c_alpha[1] + c3 * c_beta[1];
365  coeff[2] = c2 * c_alpha[2] + c3 * c_beta[2];
366  coeff[3] = c4 + c5 * c_alpha[0] + c6 * c_beta[0];
367  coeff[4] = c5 * c_alpha[1] + c6 * c_beta[1];
368  coeff[5] = c5 * c_alpha[2] + c6 * c_beta[2];
369  }
370 }

373 // Compute the mapping coefficients
374 void PAW::WarpRegion(cv::Mat_<float>& mapx, cv::Mat_<float>& mapy)
375 {
376 
377  cv::MatIterator_<float> xp = mapx.begin();
378  cv::MatIterator_<float> yp = mapy.begin();
379  cv::MatIterator_<uchar> mp = pixel_mask.begin();
380  cv::MatIterator_<int> tp = triangle_id.begin();
381 
382  // The coefficients corresponding to the current triangle
383  float * a;
384 
385  // Current triangle being processed
386  int k = -1;
387 
388  for (int y = 0; y < pixel_mask.rows; y++)
389  {
390  float yi = float(y) + min_y;
391 
392  for (int x = 0; x < pixel_mask.cols; x++)
393  {
394  float xi = float(x) + min_x;
395 
396  if (*mp == 0)
397  {
398  *xp = -1;
399  *yp = -1;
400  }
401  else
402  {
403  // triangle corresponding to the current pixel
404  int j = *tp;
405 
406  // If it is different from the previous triangle point to new coefficients
407  // This will always be the case in the first iteration, hence a will not point to nothing
408  if (j != k)
409  {
410  // Update the coefficient pointer if a new triangle is being processed
411  a = coefficients.ptr<float>(j);
412  k = j;
413  }
414 
415  //ap is now the pointer to the coefficients
416  float *ap = a;
417 
418  //look at the first coefficient (and increment). first coefficient is an x offset
419  float xo = *ap++;
420  //second coefficient is an x scale as a function of x
421  xo += *ap++ * xi;
422  //third coefficient ap(2) is an x scale as a function of y
423  *xp = float(xo + *ap++ * yi);
424 
425  //then fourth coefficient ap(3) is a y offset
426  float yo = *ap++;
427  //fifth coeff adds coeff[4]*x to y
428  yo += *ap++ * xi;
429  //final coeff adds coeff[5]*y to y
430  *yp = float(yo + *ap++ * yi);
431 
432  }
433  mp++; tp++; xp++; yp++;
434  }
435  }
436 }

442 // Is the point (x0,y0) on same side as a half-plane defined by (x1,y1), (x2, y2), and (x3, y3)
443 bool PAW::sameSide(float x0, float y0, float x1, float y1, float x2, float y2, float x3, float y3)
444 {
445 
446  float x = (x3 - x2)*(y0 - y2) - (x0 - x2)*(y3 - y2);
447  float y = (x3 - x2)*(y1 - y2) - (x1 - x2)*(y3 - y2);
448 
449  return x*y >= 0;
