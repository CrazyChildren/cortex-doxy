129 // Returns the patch expert responses given a grayscale image.
130 // Additionally returns the transform from the image coordinates to the response coordinates (and vice versa).
131 // The computation also requires the current landmark locations to compute response around, the PDM corresponding to the desired model, and the parameters describing its instance
132 // Also need to provide the size of the area of interest and the desired scale of analysis
133 void Patch_experts::Response(vector<cv::Mat_<float> >& patch_expert_responses, cv::Matx22f& sim_ref_to_img, cv::Matx22f& sim_img_to_ref, const cv::Mat_<float>& grayscale_image,
134  const PDM& pdm, const cv::Vec6f& params_global, const cv::Mat_<float>& params_local, int window_size, int scale)
135 {
136 
137  int view_id = GetViewIdx(params_global, scale);
138 
139  int n = pdm.NumberOfPoints();
140 
141  // Compute the current landmark locations (around which responses will be computed)
142  cv::Mat_<float> landmark_locations;
143 
144  pdm.CalcShape2D(landmark_locations, params_local, params_global);
145 
146  cv::Mat_<float> reference_shape;
147 
148  // Initialise the reference shape on which we'll be warping
149  cv::Vec6f global_ref(patch_scaling[scale], 0, 0, 0, 0, 0);
150 
151  // Compute the reference shape
152  pdm.CalcShape2D(reference_shape, params_local, global_ref);
153 
154  // similarity and inverse similarity transform to and from image and reference shape
155  cv::Mat_<float> reference_shape_2D = (reference_shape.reshape(1, 2).t());
156  cv::Mat_<float> image_shape_2D = landmark_locations.reshape(1, 2).t();
157 
158  sim_img_to_ref = Utilities::AlignShapesWithScale(image_shape_2D, reference_shape_2D);
159  sim_ref_to_img = sim_img_to_ref.inv(cv::DECOMP_LU);
160 
161  float a1 = sim_ref_to_img(0, 0);
162  float b1 = -sim_ref_to_img(0, 1);
163 
164  bool use_ccnf = !this->ccnf_expert_intensity.empty();
165  bool use_cen = !this->cen_expert_intensity.empty();
166 
167  // If using CCNF patch experts might need to precalculate Sigmas
168  if (use_ccnf)
169  {
170  vector<cv::Mat_<float> > sigma_components;
171 
172  // Retrieve the correct sigma component size
173  for (size_t w_size = 0; w_size < this->sigma_components.size(); ++w_size)
174  {
175  if (!this->sigma_components[w_size].empty())
176  {
177  if (window_size*window_size == this->sigma_components[w_size][0].rows)
178  {
179  sigma_components = this->sigma_components[w_size];
180  }
181  }
182  }
183 
184  // Go through all of the landmarks and compute the Sigma for each
185  for (int lmark = 0; lmark < n; lmark++)
186  {
187  // Only for visible landmarks
188  if (visibilities[scale][view_id].at<int>(lmark, 0))
189  {
190  // Precompute sigmas if they are not computed yet
191  ccnf_expert_intensity[scale][view_id][lmark].ComputeSigmas(sigma_components, window_size);
192  }
193  }
194 
195  }
196 
197  // If using CEN precalculate interpolation matrix
198  cv::Mat_<float> interp_mat;
199  if (use_cen)
200  {
201  // Assuming the same size for all experts
202  int support_region = 11;
203  int area_of_interest_width = window_size + support_region - 1;
204  int area_of_interest_height = window_size + support_region - 1;
205  int resp_size = area_of_interest_height - support_region + 1;
206  interpolationMatrix(interp_mat, resp_size, resp_size, area_of_interest_width, area_of_interest_height);
207  }
208 
209  // We do not want to create threads for invisible landmarks, so construct an index of visible ones
210  std::vector<int> vis_lmk = Collect_visible_landmarks(visibilities, scale, view_id, n);
211 
212  // calculate the patch responses for every landmark (this is the heavy lifting of landmark detection)
213  parallel_for_(cv::Range(0, vis_lmk.size()), [&](const cv::Range& range) {
214  for (int i = range.start; i < range.end; i++)
215  {
216 
217  // Work out how big the area of interest has to be to get a response of window size
218  int area_of_interest_width;
219  int area_of_interest_height;
220  int ind = vis_lmk.at(i);
221 
222  if (use_cen)
223  {
224  area_of_interest_width = window_size + cen_expert_intensity[scale][view_id][ind].width_support - 1;
225  area_of_interest_height = window_size + cen_expert_intensity[scale][view_id][ind].height_support - 1;
226  }
227  else if (use_ccnf)
228  {
229  area_of_interest_width = window_size + ccnf_expert_intensity[scale][view_id][ind].width - 1;
230  area_of_interest_height = window_size + ccnf_expert_intensity[scale][view_id][ind].height - 1;
231  }
232  else
233  {
234  area_of_interest_width = window_size + svr_expert_intensity[scale][view_id][ind].width - 1;
235  area_of_interest_height = window_size + svr_expert_intensity[scale][view_id][ind].height - 1;
236  }
237 
238  // scale and rotate to mean shape to reference frame
239  cv::Mat sim = (cv::Mat_<float>(2, 3) << a1, -b1, landmark_locations.at<float>(ind, 0) - a1 * (area_of_interest_width - 1.0f) / 2.0f + b1 * (area_of_interest_width - 1.0f) / 2.0f, b1, a1, landmark_locations.at<float>(ind + n, 0) - a1 * (area_of_interest_width - 1.0f) / 2.0f - b1 * (area_of_interest_width - 1.0f) / 2.0f);
240 
241  // Extract the region of interest around the current landmark location
242  cv::Mat_<float> area_of_interest(area_of_interest_height, area_of_interest_width, 0.0f);
243 
244  cv::warpAffine(grayscale_image, area_of_interest, sim, area_of_interest.size(), cv::WARP_INVERSE_MAP + cv::INTER_LINEAR);
245 
246  // Get intensity response either from the SVR, CCNF, or CEN patch experts (prefer CEN as they are the most accurate so far)
247  if (!cen_expert_intensity.empty())
248  {
249 
250  int im2col_size = (area_of_interest_width * area_of_interest_height - 1) / 2;
251 
252  cv::Mat_<float> prealloc_mat = preallocated_im2col[ind][im2col_size];
253 
254  // If frontal view we can do mirrored landmarks together
255  if (view_id == 0)
256  {
257  // If the patch expert does not have values, means it's a mirrored version and will be done in another part of a loop
258  if (!cen_expert_intensity[scale][view_id][ind].biases.empty())
259  {
260  // No mirrored expert, so do normally
261  int mirror_id = mirror_inds.at<int>(ind);
262  if (mirror_id == ind)
263  {
264  cv::Mat_<float> empty(0, 0, 0.0f);
265  cen_expert_intensity[scale][view_id][ind].ResponseSparse(area_of_interest, empty, patch_expert_responses[ind], empty, interp_mat, prealloc_mat, empty);
266  }
267  else
268  {
269 
270  // Grab mirrored area of interest
271 
272  // scale and rotate to mean shape to reference frame
273  cv::Mat sim_r = (cv::Mat_<float>(2, 3) << a1, -b1, landmark_locations.at<float>(mirror_id, 0) - a1 * (area_of_interest_width - 1.0f) / 2.0f + b1 * (area_of_interest_width - 1.0f) / 2.0f, b1, a1, landmark_locations.at<float>(mirror_id + n, 0) - a1 * (area_of_interest_width - 1.0f) / 2.0f - b1 * (area_of_interest_width - 1.0f) / 2.0f);
274 
275  // Extract the region of interest around the current landmark location
276  cv::Mat_<float> area_of_interest_r(area_of_interest_height, area_of_interest_width, 0.0f);
277 
278  cv::warpAffine(grayscale_image, area_of_interest_r, sim_r, area_of_interest_r.size(), cv::WARP_INVERSE_MAP + cv::INTER_LINEAR);
279 
280  cv::Mat_<float> prealloc_mat_right = preallocated_im2col[mirror_id][im2col_size];
281 
282  cen_expert_intensity[scale][view_id][ind].ResponseSparse(area_of_interest, area_of_interest_r, patch_expert_responses[ind], patch_expert_responses[mirror_id], interp_mat, prealloc_mat, prealloc_mat_right);
283 
284  preallocated_im2col[mirror_id][im2col_size] = prealloc_mat_right;
285 
286  }
287  }
288  }
289  else
290  {
291  // For space and memory saving use a mirrored patch expert
292  if (!cen_expert_intensity[scale][view_id][ind].biases.empty())
293  {
294  cv::Mat_<float> empty(0, 0, 0.0f);
295  cen_expert_intensity[scale][view_id][ind].ResponseSparse(area_of_interest, empty, patch_expert_responses[ind], empty, interp_mat, prealloc_mat, empty);
296 
297  // A slower, but slightly more accurate version
298  //cen_expert_intensity[scale][view_id][ind].Response(area_of_interest, patch_expert_responses[ind]);
299  }
300  else
301  {
302  cv::Mat_<float> empty(0, 0, 0.0f);
303  cen_expert_intensity[scale][mirror_views.at<int>(view_id)][mirror_inds.at<int>(ind)].ResponseSparse(empty, area_of_interest, empty, patch_expert_responses[ind], interp_mat, empty, prealloc_mat);
304  }
305  }
306 
307  preallocated_im2col[ind][im2col_size] = prealloc_mat;
308 
309  }
310  else if (!ccnf_expert_intensity.empty())
311  {
312  // get the correct size response window
313  patch_expert_responses[ind] = cv::Mat_<float>(window_size, window_size);
314 
315  int im2col_size = area_of_interest_width * area_of_interest_height;
316 
317  cv::Mat_<float> prealloc_mat = preallocated_im2col[ind][im2col_size];
318 
319  ccnf_expert_intensity[scale][view_id][ind].ResponseOpenBlas(area_of_interest, patch_expert_responses[ind], prealloc_mat);
320 
321  preallocated_im2col[ind][im2col_size] = prealloc_mat;
322 
323  // Below is an alternative way to compute the same, but that uses FFT instead of OpenBLAS
324  // ccnf_expert_intensity[scale][view_id][ind].Response(area_of_interest, patch_expert_responses[ind]);
325 
326  }
327  else
328  {
329  // get the correct size response window
330  patch_expert_responses[ind] = cv::Mat_<float>(window_size, window_size);
331 
332  svr_expert_intensity[scale][view_id][ind].Response(area_of_interest, patch_expert_responses[ind]);
333  }
334  }
335  });

340 // Getting the closest view center based on orientation
341 int Patch_experts::GetViewIdx(const cv::Vec6f& params_global, int scale) const
342 {
343  int idx = 0;
344 
345  float dbest;
346 
347  for(int i = 0; i < this->nViews(scale); i++)
348  {
349  float v1 = params_global[1] - centers[scale][i][0];
350  float v2 = params_global[2] - centers[scale][i][1];
351  float v3 = params_global[3] - centers[scale][i][2];
352 
353  float d = v1*v1 + v2*v2 + v3*v3;
354 
355  if(i == 0 || d < dbest)
356  {
357  dbest = d;
358  idx = i;
359  }
360  }
361  return idx;
362 }
