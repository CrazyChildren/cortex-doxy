134 void SVR_patch_expert::Response(const cv::Mat_<float>& area_of_interest, cv::Mat_<float>& response)
135 {
136 
137  int response_height = area_of_interest.rows - weights.rows + 1;
138  int response_width = area_of_interest.cols - weights.cols + 1;
139 
140  // the patch area on which we will calculate reponses
141  cv::Mat_<float> normalised_area_of_interest;
142 
143  if(response.rows != response_height || response.cols != response_width)
144  {
145  response.create(response_height, response_width);
146  }
147 
148  // If type is raw just normalise mean and standard deviation
149  if(type == 0)
150  {
151  // Perform normalisation across whole patch
152  cv::Scalar mean;
153  cv::Scalar std;
154 
155  cv::meanStdDev(area_of_interest, mean, std);
156  // Avoid division by zero
157  if(std[0] == 0)
158  {
159  std[0] = 1;
160  }
161  normalised_area_of_interest = (area_of_interest - mean[0]) / std[0];
162  }
163  // If type is gradient, perform the image gradient computation
164  else if(type == 1)
165  {
166  Grad(area_of_interest, normalised_area_of_interest);
167  }
168  else
169  {
170  printf("ERROR(%s,%d): Unsupported patch type %d!\n", __FILE__,__LINE__, type);
171  abort();
172  }
173 
174  cv::Mat_<float> svr_response;
175 
176  // The empty matrix as we don't pass precomputed dft's of image
177  cv::Mat_<double> empty_matrix_0(0,0,0.0);
178  cv::Mat_<float> empty_matrix_1(0,0,0.0);
179  cv::Mat_<float> empty_matrix_2(0,0,0.0);
180 
181  // Efficient calc of patch expert SVR response across the area of interest
182  matchTemplate_m(normalised_area_of_interest, empty_matrix_0, empty_matrix_1, empty_matrix_2, weights, weights_dfts, svr_response, cv::TM_CCOEFF_NORMED);
183 
184  response.create(svr_response.size());
185  cv::MatIterator_<float> p = response.begin();
186 
187  cv::MatIterator_<float> q1 = svr_response.begin(); // respone for each pixel
188  cv::MatIterator_<float> q2 = svr_response.end();
189 
190  while(q1 != q2)
191  {
192  // the SVR response passed into logistic regressor
193  *p++ = 1.0/(1.0 + exp( -(*q1++ * scaling + bias )));
194  }
