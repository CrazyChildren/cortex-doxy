874 // 把脸分成左右两半，均衡两边光线，这一算法效率可以改进
875 // Histogram Equalizae seperately for the left and right sides of the face,
876 // so that if there is a strong light on one side but not the other, it will still look OK.
877 void PreprocessFace::equalizeLeftAndRightHalves(Mat &faceImg)
878 {
879  MyDebug;
880  // It is common that there is stronger light from one half of the face than the other. In that case,
881  // if you simply did histogram equalization on the whole face then it would make one half dark and
882  // one half bright. So we will do histogram equalization separately on each face half, so they will
883  // both look similar on average. But this would cause a sharp edge in the middle of the face, because
884  // the left half and right half would be suddenly different. So we also histogram equalize the whole
885  // image, and in the middle part we blend the 3 images together for a smooth brightness transition.
886 
887  int w = faceImg.cols;
888  int h = faceImg.rows;
889 
890  // 1) First, equalize the whole face.
891  Mat wholeFace;
892  equalizeHist(faceImg, wholeFace);
893 
894  // 2) Equalize the left half and the right half of the face separately.
895  int midX = w/2;
896  Mat leftSide = faceImg(Rect(0,0, midX,h));
897  Mat rightSide = faceImg(Rect(midX,0, w-midX,h));
898  equalizeHist(leftSide, leftSide);
899  equalizeHist(rightSide, rightSide);
900 
901  // 3) Combine the left half and right half and whole face together, so that it has a smooth transition.
902  for (int y=0; y<h; y++) {
903  for (int x=0; x<w; x++) {
904  int v;
905  if (x < w/4) { // Left 25%: just use the left face.
906  v = leftSide.at<uchar>(y,x);
907  }
908  else if (x < w*2/4) { // Mid-left 25%: blend the left face & whole face.
909  int lv = leftSide.at<uchar>(y,x);
910  int wv = wholeFace.at<uchar>(y,x);
911  // Blend more of the whole face as it moves further right along the face.
912  float f = (x - w*1/4) / (float)(w*0.25f);
913  v = cvRound((1.0f - f) * lv + (f) * wv);
914  }
915  else if (x < w*3/4) { // Mid-right 25%: blend the right face & whole face.
916  int rv = rightSide.at<uchar>(y,x-midX);
917  int wv = wholeFace.at<uchar>(y,x);
918  // Blend more of the right-side face as it moves further right along the face.
919  float f = (x - w*2/4) / (float)(w*0.25f);
920  v = cvRound((1.0f - f) * wv + (f) * rv);
921  }
922  else { // Right 25%: just use the right face.
923  v = rightSide.at<uchar>(y,x-midX);
924  }
925  faceImg.at<uchar>(y,x) = v;
926  }// end x loop
927  }//end y loop
928 }
