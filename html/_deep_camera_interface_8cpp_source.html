301 void DeepCameraInterface::removeBackground(rs2::video_frame &color_frame, const rs2::depth_frame &depth_frame, float depth_scale, float clipping_dist)
302 {
303  const uint16_t* p_depth_frame = reinterpret_cast<const uint16_t*>(depth_frame.get_data());
304  uint8_t* p_color_frame = reinterpret_cast<uint8_t*>(const_cast<void*>(color_frame.get_data()));
305 
306  int width = color_frame.get_width();
307  int height = color_frame.get_height();
308  int color_bpp = color_frame.get_bytes_per_pixel();
309 
310  #pragma omp parallel for schedule(dynamic) //Using OpenMP to try to parallelise the loop
311  for (int y = 0; y < height; y++)
312  {
313  auto depth_pixel_index = y * width;
314  for (int x = 0; x < width; x++, ++depth_pixel_index)
315  {
316  // Get the depth value of the current pixel
317  auto pixels_distance = depth_scale * p_depth_frame[depth_pixel_index];
318 
319  // Check if the depth value is invalid (<=0) or greater than the threashold
320  if (pixels_distance <= 0.f || pixels_distance > clipping_dist)
321  {
322  // Calculate the offset in color frame's buffer to current pixel
323  auto offset = depth_pixel_index * color_bpp;
324 
325  // Set pixel to "background" color (0x999999)
326  std::memset(&p_color_frame[offset], 0x99, color_bpp);
327  }
328  }
329  }
