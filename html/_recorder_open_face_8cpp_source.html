59 void CreateDirectory(std::string output_path)
60 {
61  // Removing trailing separators, as that causes issues with directory creation in unix
62  while (output_path[output_path.size() - 1] == '/' || output_path[output_path.size() - 1] == '\\')
63  {
64  output_path = output_path.substr(0, output_path.size() - 1);
65  }
66 
67  // Creating the right directory structure
68  auto p = path(output_path);
69 
70  if (!boost::filesystem::exists(p))
71  {
72  bool success = boost::filesystem::create_directories(p);
73 
74  if (!success)
75  {
76  std::cout << "ERROR: failed to create output directory:" << p.string() << ", do you have permission to create directory" << std::endl;
77  exit(1);
78  }

220 RecorderOpenFace::RecorderOpenFace(const std::string in_filename, const RecorderOpenFaceParameters& parameters, std::vector<std::string>& arguments):video_writer(), params(parameters)
221 {
222 
223  // From the filename, strip out the name without directory and extension
224  if (boost::filesystem::is_directory(in_filename))
225  {
226  out_name = boost::filesystem::canonical(boost::filesystem::path(in_filename)).filename().string();
227  }
228  else
229  {
230  out_name = boost::filesystem::path(in_filename).filename().replace_extension("").string();
231  }
232 
233  // Consuming the input arguments
234  bool* valid = new bool[arguments.size()];
235 
236  for (size_t i = 0; i < arguments.size(); ++i)
237  {
238  valid[i] = true;
239  }
240 
241  for (size_t i = 0; i < arguments.size(); ++i)
242  {
243  if (arguments[i].compare("-out_dir") == 0)
244  {
245  record_root = arguments[i + 1];
246  }
247  }
248 
249  // Determine output directory
250  bool output_found = false;
251  for (size_t i = 0; i < arguments.size(); ++i)
252  {
253  if (!output_found && arguments[i].compare("-of") == 0)
254  {
255  record_root = (boost::filesystem::path(record_root) / boost::filesystem::path(arguments[i + 1])).remove_filename().string();
256  out_name = path(boost::filesystem::path(arguments[i + 1])).replace_extension("").filename().string();
257  valid[i] = false;
258  valid[i + 1] = false;
259  i++;
260  output_found = true;
261  }
262  }
263 
264  // If recording directory not set, record to default location
265  if (record_root.empty())
266  record_root = default_record_directory;
267 
268  for (int i = (int)arguments.size() - 1; i >= 0; --i)
269  {
270  if (!valid[i])
271  {
272  arguments.erase(arguments.begin() + i);
273  }
274  }
275 
276  PrepareRecording(in_filename);
277 
278 }

316 void RecorderOpenFace::WriteObservation()
317 {
318 
319  // Write out the CSV file (it will always be there, even if not outputting anything more but frame/face numbers)
320  if(!csv_recorder.isOpen())
321  {
322  // As we are writing out the header, work out some things like number of landmarks, names of AUs etc.
323  int num_face_landmarks = landmarks_2D.rows / 2;
324  int num_eye_landmarks = (int)eye_landmarks2D.size();
325  int num_model_modes = pdm_params_local.rows;
326 
327  std::vector<std::string> au_names_class;
328  for (auto au : au_occurences)
329  {
330  au_names_class.push_back(au.first);
331  }
332 
333  std::sort(au_names_class.begin(), au_names_class.end());
334 
335  std::vector<std::string> au_names_reg;
336  for (auto au : au_intensities)
337  {
338  au_names_reg.push_back(au.first);
339  }
340 
341  std::sort(au_names_reg.begin(), au_names_reg.end());
342 
343  metadata_file << "Output csv:" << csv_filename << endl;
344  metadata_file << "Gaze: " << params.outputGaze() << endl;
345  metadata_file << "AUs: " << params.outputAUs() << endl;
346  metadata_file << "Landmarks 2D: " << params.output2DLandmarks() << endl;
347  metadata_file << "Landmarks 3D: " << params.output3DLandmarks() << endl;
348  metadata_file << "Pose: " << params.outputPose() << endl;
349  metadata_file << "Shape parameters: " << params.outputPDMParams() << endl;
350 
351  csv_filename = (path(record_root) / csv_filename).string();
352  csv_recorder.Open(csv_filename, params.isSequence(), params.output2DLandmarks(), params.output3DLandmarks(), params.outputPDMParams(), params.outputPose(),
353  params.outputAUs(), params.outputGaze(), num_face_landmarks, num_model_modes, num_eye_landmarks, au_names_class, au_names_reg);
354  }
355 
356  this->csv_recorder.WriteLine(face_id, frame_number, timestamp, landmark_detection_success,
357  landmark_detection_confidence, landmarks_2D, landmarks_3D, pdm_params_local, pdm_params_global, head_pose,
358  gaze_direction0, gaze_direction1, gaze_angle, eye_landmarks2D, eye_landmarks3D, au_intensities, au_occurences);
359 
360  if(params.outputHOG())
361  {
362  this->hog_recorder.Write();
363  }
364 
365  // Write aligned faces
366  if (params.outputAlignedFaces())
367  {
368 
369  if (!aligned_writing_thread_started)
370  {
371  aligned_writing_thread_started = true;
372  int capacity = (1024 * 1024 * ALIGNED_QUEUE_CAPACITY) / (aligned_face.size().width *aligned_face.size().height * aligned_face.channels()) + 1;
373  aligned_face_queue.set_capacity(capacity);
374 
375  // Start the alignment output thread
376  aligned_writing_thread = std::thread(&RecorderOpenFace::AlignedImageWritingTask, this);
377  }
378 
379  char name[100];
380 
381  // Filename is based on frame number (TODO stringstream this)
382  if(params.isSequence())
383  std::sprintf(name, "frame_det_%02d_%06d.", face_id, frame_number);
384  else
385  std::sprintf(name, "face_det_%06d.", face_id);
386 
387  // Construct the output filename
388  boost::filesystem::path slash("/");
389 
390  std::string preferredSlash = slash.make_preferred().string();
391 
392  string out_file = aligned_output_directory + preferredSlash + string(name) + params.imageFormatAligned();
393 
394  if(params.outputBadAligned() || landmark_detection_success)
395  {
396  aligned_face_queue.push(std::pair<std::string, cv::Mat>(out_file, aligned_face));
