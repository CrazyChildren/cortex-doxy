250 bool CLNF::Read_CLNF(string clnf_location)
251 {
252  // Location of modules
253  ifstream locations(clnf_location.c_str(), ios_base::in);
254 
255  if(!locations.is_open())
256  {
257  cout << "Couldn't open the CLNF model file aborting" << endl;
258  cout.flush();
259  return false;
260  }
261 
262  string line;
263 
264  vector<string> intensity_expert_locations;
265  vector<string> ccnf_expert_locations;
266  vector<string> cen_expert_locations;
267  string early_term_loc;
268 
269  // The other module locations should be defined as relative paths from the main model
270  boost::filesystem::path root = boost::filesystem::path(clnf_location).parent_path();
271 
272  // The main file contains the references to other files
273  while (!locations.eof())
274  {
275 
276  getline(locations, line);
277 
278  stringstream lineStream(line);
279 
280  string module;
281  string location;
282 
283  // figure out which module is to be read from which file
284  lineStream >> module;
285 
286  getline(lineStream, location);
287 
288  if(location.size() > 0)
289  location.erase(location.begin()); // remove the first space
290 
291  // remove carriage return at the end for compatibility with unix systems
292  if(location.size() > 0 && location.at(location.size()-1) == '\r')
293  {
294  location = location.substr(0, location.size()-1);
295  }
296 
297  // append the lovstion to root location (boost syntax)
298  location = (root / location).string();
299 
300  if (module.compare("PDM") == 0)
301  {
302  cout << "Reading the PDM module from: " << location << "....";
303  bool read_success = pdm.Read(location);
304 
305  if (!read_success)
306  {
307  return false;
308  }
309 
310  cout << "Done" << endl;
311  }
312  else if (module.compare("Triangulations") == 0)
313  {
314  cout << "Reading the Triangulations module from: " << location << "....";
315  ifstream triangulationFile(location.c_str(), ios_base::in);
316 
317  if(!triangulationFile.is_open())
318  {
319  return false;
320  }
321 
322  LandmarkDetector::SkipComments(triangulationFile);
323 
324  int numViews;
325  triangulationFile >> numViews;
326 
327  // read in the triangulations
328  triangulations.resize(numViews);
329 
330  for(int i = 0; i < numViews; ++i)
331  {
332  LandmarkDetector::SkipComments(triangulationFile);
333  LandmarkDetector::ReadMat(triangulationFile, triangulations[i]);
334  }
335  cout << "Done" << endl;
336  }
337  else if(module.compare("PatchesIntensity") == 0)
338  {
339  intensity_expert_locations.push_back(location);
340  }
341  else if(module.compare("PatchesCCNF") == 0)
342  {
343  ccnf_expert_locations.push_back(location);
344  }
345  else if (module.compare("PatchesCEN") == 0)
346  {
347  cen_expert_locations.push_back(location);
348  }
349  else if (module.compare("EarlyTermination") == 0)
350  {
351  early_term_loc = location;
352  }
353  }
354 
355  // Initialise the patch experts
356  bool read_success = patch_experts.Read(intensity_expert_locations, ccnf_expert_locations, cen_expert_locations, early_term_loc);
357 
358  if(!read_success)
359  {
360  return false;
361  }
362 
363  return true;
364

627 // The main internal landmark detection call (should not be used externally?)
628 bool CLNF::DetectLandmarks(const cv::Mat_<uchar> &image, FaceModelParameters& params)
629 {
630 
631  // TODO this could be moved out
632  cv::Mat_<float> gray_image_flt;
633  image.convertTo(gray_image_flt, CV_32F);
634 
635  // Fits from the current estimate of local and global parameters in the model
636  bool fit_success = Fit(gray_image_flt, params.window_sizes_current, params);
637 
638  // Store the landmarks converged on in detected_landmarks
639  pdm.CalcShape2D(detected_landmarks, params_local, params_global);
640 
641  if(params.refine_hierarchical && hierarchical_models.size() > 0)
642  {
643  bool parts_used = false;
644 
645  // Do the hierarchical models in parallel
646  parallel_for_(cv::Range(0, hierarchical_models.size()), [&](const cv::Range& range) {
647  for (int part_model = range.start; part_model < range.end; part_model++)
648  {
649 
650  int n_part_points = hierarchical_models[part_model].pdm.NumberOfPoints();
651 
652  vector<pair<int, int>> mappings = this->hierarchical_mapping[part_model];
653 
654  cv::Mat_<float> part_model_locs(n_part_points * 2, 1, 0.0f);
655 
656  // Extract the corresponding landmarks
657  for (size_t mapping_ind = 0; mapping_ind < mappings.size(); ++mapping_ind)
658  {
659  part_model_locs.at<float>(mappings[mapping_ind].second) = detected_landmarks.at<float>(mappings[mapping_ind].first);
660  part_model_locs.at<float>(mappings[mapping_ind].second + n_part_points) = detected_landmarks.at<float>(mappings[mapping_ind].first + this->pdm.NumberOfPoints());
661  }
662 
663  // Fit the part based model PDM

669  parts_used = true;
670

672 
673  // Do the actual landmark detection

676  }
677  else
678  {
679  hierarchical_models[part_model].pdm.CalcShape2D(hierarchical_models[part_model].detected_landmarks, hierarchical_models[part_model].params_local, hierarchical_models[part_model].params_global);
680  }
681 
682  }
683  });
684

686  if(parts_used)
687  {
688 
689  for (size_t part_model = 0; part_model < hierarchical_models.size(); ++part_model)
690  {
691  vector<pair<int, int>> mappings = this->hierarchical_mapping[part_model];

698  }

702  pdm.CalcShape2D(detected_landmarks, params_local, params_global);
703  }
704 
705  }

710 
711  cv::Vec3d orientation(params_global[1], params_global[2], params_global[3]);
712 
713  detection_certainty = landmark_validator.Check(orientation, image, detected_landmarks);
714 
715  detection_success = detection_certainty > params.validation_boundary;
716 
717  }

719  {
720  detection_success = fit_success;
721  if(fit_success)
722  {
723  detection_certainty = 1;
724  }
725  else
726  {
727  detection_certainty = 0;
728  }
729 
730  }

736 bool CLNF::Fit(const cv::Mat_<float>& im, const std::vector<int>& window_sizes, const FaceModelParameters& parameters)
737 {
738  // Making sure it is a single channel image
739  assert(im.channels() == 1);
740 
741  // Placeholder for the landmarks
742  cv::Mat_<float> current_shape(2 * pdm.NumberOfPoints() , 1, 0.0f);
743 
744  int n = pdm.NumberOfPoints();
745 
746  int num_scales = patch_experts.patch_scaling.size();
747 
748  // Storing the patch expert response maps
749  vector<cv::Mat_<float> > patch_expert_responses(n);
750 
751  // Converting from image space to patch expert space (normalised for rotation and scale)
752  cv::Matx22f sim_ref_to_img;
753  cv::Matx22f sim_img_to_ref;
754 
755  FaceModelParameters tmp_parameters = parameters;
756 
757  // Active scale is there in case we need to upsample too much
758  int active_scale = 0;
759 
760  // Optimise the model across a number of areas of interest (usually in descending window size and ascending scale size)
761  for(int scale = 0; scale < num_scales; scale++)
762  {
763  // Control the number of iterations through window size
764  if (window_sizes[scale] == 0)
765  continue;
766 
767  int window_size = window_sizes[scale];
768 
769  // The patch expert response computation
770  patch_experts.Response(patch_expert_responses, sim_ref_to_img, sim_img_to_ref, im, pdm, params_global, params_local, window_size, scale);
771 
772  if(parameters.refine_parameters == true)
773  {
774  int scale_max = scale >= 2 ? 2 : scale;
775 
776  // Adapt the parameters based on scale (wan't to reduce regularisation as scale increases, but increase sigma and Tikhonov)
777  tmp_parameters.reg_factor = parameters.reg_factor - 15 * log(patch_experts.patch_scaling[scale_max]/0.25)/log(2);
778 
779  if(tmp_parameters.reg_factor <= 0)
780  tmp_parameters.reg_factor = 0.001;
781 
782  tmp_parameters.sigma = parameters.sigma + 0.25 * log(patch_experts.patch_scaling[scale_max]/0.25)/log(2);
783  tmp_parameters.weight_factor = parameters.weight_factor + 2 * parameters.weight_factor * log(patch_experts.patch_scaling[scale_max]/0.25)/log(2);
784  }
785 
786  // Get the current landmark locations
787  pdm.CalcShape2D(current_shape, params_local, params_global);
788 
789  // Get the view used by patch experts
790  int view_id = patch_experts.GetViewIdx(params_global, scale);
791  this->view_used = view_id;
792 
793  // the actual optimisation step
794  this->NU_RLMS(params_global, params_local, patch_expert_responses, cv::Vec6f(params_global), params_local.clone(), current_shape, sim_img_to_ref, sim_ref_to_img, window_size, view_id, true, scale, this->landmark_likelihoods, tmp_parameters, false);
795 
796  // non-rigid optimisation
797 
798  // If we are terminating next iteration, make sure to record the model likelihood
799  if(scale == num_scales - 1 || window_sizes[scale + 1] == 0 || params_global[0] < 0.30)
800  {
801  this->model_likelihood = this->NU_RLMS(params_global, params_local, patch_expert_responses, cv::Vec6f(params_global), params_local.clone(), current_shape, sim_img_to_ref, sim_ref_to_img, window_size, view_id, false, scale, this->landmark_likelihoods, tmp_parameters, true);
802  }
803  else
804  {
805  this->NU_RLMS(params_global, params_local, patch_expert_responses, cv::Vec6f(params_global), params_local.clone(), current_shape, sim_img_to_ref, sim_ref_to_img, window_size, view_id, false, scale, this->landmark_likelihoods, tmp_parameters, false);
806  }
807 
808  // Can't track very small images reliably (less than ~30px across)
809  if (params_global[0] < 0.25)
810  {
811  cout << "Face too small for landmark detection" << endl;
812  return false;
813  }
814 
815  // Making sure we do not upsample too much
816  if (active_scale < num_scales - 1 && 0.9 * patch_experts.patch_scaling[active_scale + 1] < params_global[0])
817  active_scale = active_scale + 1;
818 
819  }
820 
821  return true;
822 }

824 void CLNF::NonVectorisedMeanShift_precalc_kde(cv::Mat_<float>& out_mean_shifts, const vector<cv::Mat_<float> >& patch_expert_responses, const cv::Mat_<float> &dxs, const cv::Mat_<float> &dys, int resp_size, float a, int scale, int view_id, map<int, cv::Mat_<float> >& kde_resp_precalc)
825 {
826 
827  int n = dxs.rows;
828 
829  cv::Mat_<float> kde_resp;
830  float step_size = 0.1;
831 
832  // if this has not been precomputer, precompute it, otherwise use it
833  if(kde_resp_precalc.find(resp_size) == kde_resp_precalc.end())
834  {
835  kde_resp = cv::Mat_<float>((int)((resp_size / step_size)*(resp_size/step_size)), resp_size * resp_size);
836  cv::MatIterator_<float> kde_it = kde_resp.begin();
837 
838  for(int x = 0; x < resp_size/step_size; x++)
839  {
840  float dx = x * step_size;
841  for(int y = 0; y < resp_size/step_size; y++)
842  {
843  float dy = y * step_size;
844 
845  int ii,jj;
846  float v,vx,vy;
847 
848  for(ii = 0; ii < resp_size; ii++)
849  {
850  vx = (dy-ii)*(dy-ii);
851  for(jj = 0; jj < resp_size; jj++)
852  {
853  vy = (dx-jj)*(dx-jj);
854 
855  // the KDE evaluation of that point
856  v = exp(a*(vx+vy));
857 
858  *kde_it++ = v;
859  }
860  }
861  }
862  }
863 
864  kde_resp_precalc[resp_size] = kde_resp.clone();
865  }
866  else
867  {
868  // use the precomputed version
869  kde_resp = kde_resp_precalc.find(resp_size)->second;
870  }
871 
872  // for every point (patch) calculating mean-shift
873  for(int i = 0; i < n; i++)
874  {
875  if(patch_experts.visibilities[scale][view_id].at<int>(i,0) == 0)
876  {
877  out_mean_shifts.at<float>(i,0) = 0;
878  out_mean_shifts.at<float>(i+n,0) = 0;
879  continue;
880  }
881 
882  // indices of dx, dy
883  float dx = dxs.at<float>(i);
884  float dy = dys.at<float>(i);
885 
886  // Ensure that we are within bounds (important for precalculation)
887  if(dx < 0)
888  dx = 0;
889  if(dy < 0)
890  dy = 0;
891  if(dx > resp_size - step_size)
892  dx = resp_size - step_size;
893  if(dy > resp_size - step_size)
894  dy = resp_size - step_size;
895 
896  // Pick the row from precalculated kde that approximates the current dx, dy best
897  int closest_col = (int)(dy /step_size + 0.5); // Plus 0.5 is there, as C++ rounds down with int cast
898  int closest_row = (int)(dx /step_size + 0.5); // Plus 0.5 is there, as C++ rounds down with int cast
899 
900  int idx = closest_row * ((int)(resp_size/step_size + 0.5)) + closest_col; // Plus 0.5 is there, as C++ rounds down with int cast
901 
902  cv::MatIterator_<float> kde_it = kde_resp.begin() + kde_resp.cols*idx;
903 
904  float mx=0.0;
905  float my=0.0;
906  float sum=0.0;
907 
908  // Iterate over the patch responses here
909  cv::MatConstIterator_<float> p = patch_expert_responses[i].begin();
910 
911  // TODO maybe do through MatMuls instead?
912  for(int ii = 0; ii < resp_size; ii++)
913  {
914  for(int jj = 0; jj < resp_size; jj++)
915  {
916 
917  // the KDE evaluation of that point multiplied by the probability at the current, xi, yi
918  float v = (*p++) * (*kde_it++);
919 
920  sum += v;
921 
922  // mean shift in x and y
923  mx += v*jj;
924  my += v*ii;
925 
926  }
927  }
928 
929  float msx = (mx/sum - dx);
930  float msy = (my/sum - dy);
931 
932  out_mean_shifts.at<float>(i,0) = msx;
933  out_mean_shifts.at<float>(i+n,0) = msy;
934 
935  }
936

991 //=============================================================================
992 float CLNF::NU_RLMS(cv::Vec6f& final_global, cv::Mat_<float>& final_local, const vector<cv::Mat_<float> >& patch_expert_responses, const cv::Vec6f& initial_global, const cv::Mat_<float>& initial_local,
993  const cv::Mat_<float>& base_shape, const cv::Matx22f& sim_img_to_ref, const cv::Matx22f& sim_ref_to_img, int resp_size, int view_id, bool rigid, int scale, cv::Mat_<float>& landmark_lhoods,
994  const FaceModelParameters& parameters, bool compute_lhood)
995 {
996 
997  int n = pdm.NumberOfPoints();
998 
999  // Mean, eigenvalues, eigenvectors
1000  cv::Mat_<float> M = this->pdm.mean_shape;
1001  cv::Mat_<float> E = this->pdm.eigen_values;
1002  //Mat_<float> V = this->pdm.princ_comp;
1003 
1004  int m = pdm.NumberOfModes();
1005 
1006  cv::Vec6f current_global(initial_global);
1007 
1008  cv::Mat_<float> current_local = initial_local.clone();
1009 
1010  cv::Mat_<float> current_shape;
1011  cv::Mat_<float> previous_shape;
1012 
1013  // Pre-calculate the regularisation term
1014  cv::Mat_<float> regTerm;
1015 
1016  if(rigid)
1017  {
1018  regTerm = cv::Mat_<float>::zeros(6,6);
1019  }
1020  else
1021  {
1022  cv::Mat_<float> regularisations = cv::Mat_<float>::zeros(1, 6 + m);
1023 
1024  // Setting the regularisation to the inverse of eigenvalues
1025  cv::Mat(parameters.reg_factor / E).copyTo(regularisations(cv::Rect(6, 0, m, 1)));
1026  regTerm = cv::Mat::diag(regularisations.t());
1027  }
1028 
1029  cv::Mat_<float> WeightMatrix;
1030  GetWeightMatrix(WeightMatrix, scale, view_id, parameters);
1031 
1032  cv::Mat_<float> dxs, dys;
1033 
1034  // The preallocated memory for the mean shifts
1035  cv::Mat_<float> mean_shifts(2 * pdm.NumberOfPoints(), 1, 0.0);
1036 
1037  // Number of iterations
1038  for(int iter = 0; iter < parameters.num_optimisation_iteration; iter++)
1039  {
1040  // get the current estimates of x
1041  pdm.CalcShape2D(current_shape, current_local, current_global);
1042 
1043  if(iter > 0)
1044  {
1045  // if the shape hasn't changed terminate
1046  if(norm(current_shape, previous_shape) < 0.01)
1047  {
1048  break;
1049  }
1050  }
1051 
1052  current_shape.copyTo(previous_shape);
1053 
1054  // Jacobian, and transposed weighted jacobian
1055  cv::Mat_<float> J, J_w_t;
1056 
1057  // calculate the appropriate Jacobians in 2D, even though the actual behaviour is in 3D, using small angle approximation and oriented shape
1058  if(rigid)
1059  {
1060  pdm.ComputeRigidJacobian(current_local, current_global, J, WeightMatrix, J_w_t);
1061  }
1062  else
1063  {
1064  pdm.ComputeJacobian(current_local, current_global, J, WeightMatrix, J_w_t);
1065  }
1066 
1067  // useful for mean shift calculation
1068  float a = -0.5/(parameters.sigma * parameters.sigma);
1069 
1070  cv::Mat_<float> current_shape_2D = current_shape.reshape(1, 2).t();
1071  cv::Mat_<float> base_shape_2D = base_shape.reshape(1, 2).t();
1072 
1073  cv::Mat_<float> offsets;
1074  cv::Mat((current_shape_2D - base_shape_2D) * cv::Mat(sim_img_to_ref).t()).convertTo(offsets, CV_32F);
1075 
1076  dxs = offsets.col(0) + (resp_size-1)/2;
1077  dys = offsets.col(1) + (resp_size-1)/2;
1078 
1079  NonVectorisedMeanShift_precalc_kde(mean_shifts, patch_expert_responses, dxs, dys, resp_size, a, scale, view_id, kde_resp_precalc);
1080 
1081  // Now transform the mean shifts to the the image reference frame, as opposed to one of ref shape (object space)
1082  cv::Mat_<float> mean_shifts_2D = (mean_shifts.reshape(1, 2)).t();
1083 
1084  mean_shifts_2D = mean_shifts_2D * cv::Mat(sim_ref_to_img).t();
1085  mean_shifts = cv::Mat(mean_shifts_2D.t()).reshape(1, n*2);
1086 
1087  // remove non-visible observations
1088  for(int i = 0; i < n; ++i)
1089  {
1090  // if patch unavailable for current index
1091  if(patch_experts.visibilities[scale][view_id].at<int>(i,0) == 0)
1092  {
1093  cv::Mat Jx = J.row(i);
1094  Jx = cvScalar(0);
1095  cv::Mat Jy = J.row(i+n);
1096  Jy = cvScalar(0);
1097 
1098  Jx = J_w_t.col(i);
1099  Jx = cvScalar(0);
1100  Jy = J_w_t.col(i + n);
1101  Jy = cvScalar(0);
1102 
1103  mean_shifts.at<float>(i,0) = 0.0f;
1104  mean_shifts.at<float>(i+n,0) = 0.0f;
1105  }
1106  }
1107 
1108  // projection of the meanshifts onto the jacobians (using the weighted Jacobian, see Baltrusaitis 2013)
1109  cv::Mat_<float> J_w_t_m = J_w_t * mean_shifts;
1110 
1111  // Add the regularisation term
1112  if(!rigid)
1113  {
1114  J_w_t_m(cv::Rect(0,6,1, m)) = J_w_t_m(cv::Rect(0,6,1, m)) - regTerm(cv::Rect(6,6, m, m)) * current_local;
1115  }
1116 
1117  cv::Mat_<float> Hessian = regTerm.clone();
1118 
1119  // Perform matrix multiplication in OpenBLAS (fortran call)
1120  float alpha1 = 1.0;
1121  float beta1 = 1.0;
1122  char N[2]; N[0] = 'N';
1123  sgemm_(N, N, &J.cols, &J_w_t.rows, &J_w_t.cols, &alpha1, (float*)J.data, &J.cols, (float*)J_w_t.data, &J_w_t.cols, &beta1, (float*)Hessian.data, &J.cols);
1124 
1125  // Above is a fast (but ugly) version of
1126  // cv::Mat_<float> Hessian = J_w_t * J + regTerm;
1127 
1128  // Solve for the parameter update (from Baltrusaitis 2013 based on eq (36) Saragih 2011)
1129  cv::Mat_<float> param_update;
1130  cv::solve(Hessian, J_w_t_m, param_update, cv::DECOMP_CHOLESKY);
1131 
1132  // update the reference
1133  pdm.UpdateModelParameters(param_update, current_local, current_global);
1134 
1135  // clamp to the local parameters for valid expressions
1136  pdm.Clamp(current_local, current_global, parameters);
1137 
1138  }
1139 
1140  // compute the log likelihood
1141  float loglhood = 0;
1142 
1143  if(compute_lhood)
1144  {
1145  landmark_lhoods = cv::Mat_<float>(n, 1, -1e8);
1146 
1147  for(int i = 0; i < n; i++)
1148  {
1149 
1150  if(patch_experts.visibilities[scale][view_id].at<int>(i,0) == 0 )
1151  {
1152  continue;
1153  }
1154  float dx = dxs.at<float>(i);
1155  float dy = dys.at<float>(i);
1156 
1157  int ii,jj;
1158  float v,vx,vy,sum=0.0;
1159 
1160  // Iterate over the patch responses here
1161  cv::MatConstIterator_<float> p = patch_expert_responses[i].begin();
1162 
1163  for(ii = 0; ii < resp_size; ii++)
1164  {
1165  vx = (dy-ii)*(dy-ii);
1166  for(jj = 0; jj < resp_size; jj++)
1167  {
1168  vy = (dx-jj)*(dx-jj);
1169 
1170  // the probability at the current, xi, yi
1171  v = *p++;
1172 
1173  // the KDE evaluation of that point
1174  v *= exp(-0.5*(vx+vy)/(parameters.sigma * parameters.sigma));
1175 
1176  sum += v;
1177  }
1178  }
1179  landmark_lhoods.at<float>(i,0) = sum;
1180 
1181  // the offset is there for numerical stability
1182  loglhood += log(sum + 1e-8);
1183 
1184  }
1185  loglhood = loglhood/sum(patch_experts.visibilities[scale][view_id])[0];
1186  }
1187 
1188  final_global = current_global;
1189  final_local = current_local;
1190 
1191  return loglhood;
1192

1196 cv::Mat_<float> CLNF::GetShape(float fx, float fy, float cx, float cy) const
1197 {

1201  this->pdm.CalcShape3D(shape3d, this->params_local);

1204

1223  float Y = Z * ((this->detected_landmarks.at<float>(i + n) - cy)/fy);
1224

1226  outShape.at<float>(i,1) = Y;
1227  outShape.at<float>(i,2) = Z;
1228 
1229  }
1230

1232  return outShape.t();
