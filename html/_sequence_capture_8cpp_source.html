350 bool SequenceCapture::OpenImageSequence(std::string directory, float fx, float fy, float cx, float cy)
351 {
352  INFO_STREAM("Attempting to read from directory: " << directory);
353 
354  no_input_specified = false;
355  frame_num = 0;
356  time_stamp = 0;
357 
358  image_files.clear();
359 
360  boost::filesystem::path image_directory(directory);
361 
362  if (!boost::filesystem::exists(image_directory))
363  {
364  std::cout << "Provided directory does not exist: " << directory << std::endl;
365  return false;
366  }
367 
368  std::vector<boost::filesystem::path> file_in_directory;
369  copy(boost::filesystem::directory_iterator(image_directory), boost::filesystem::directory_iterator(), back_inserter(file_in_directory));
370 
371  // Sort the images in the directory first
372  sort(file_in_directory.begin(), file_in_directory.end());
373 
374  std::vector<std::string> curr_dir_files;
375 
376  for (std::vector<boost::filesystem::path>::const_iterator file_iterator(file_in_directory.begin()); file_iterator != file_in_directory.end(); ++file_iterator)
377  {
378  // Possible image extension .jpg and .png
379  if (file_iterator->extension().string().compare(".jpg") == 0 || file_iterator->extension().string().compare(".jpeg") == 0 || file_iterator->extension().string().compare(".png") == 0 || file_iterator->extension().string().compare(".bmp") == 0)
380  {
381  curr_dir_files.push_back(file_iterator->string());
382  }
383  }
384 
385  image_files = curr_dir_files;
386 
387  if (image_files.empty())
388  {
389  std::cout << "No images found in the directory: " << directory << std::endl;
390  return false;
391  }
392 
393  // Assume all images are same size in an image sequence
394  cv::Mat tmp = cv::imread(image_files[0], cv::IMREAD_COLOR);
395  this->frame_height = tmp.size().height;
396  this->frame_width = tmp.size().width;
397 
398  SetCameraIntrinsics(fx, fy, cx, cy);
399 
400  // No fps as we have a sequence
401  this->fps = 0;
402 
403  this->name = directory;
404 
405  is_webcam = false;
406  is_image_seq = true;
407  vid_length = image_files.size();
408  capturing = true;
409

416 void SequenceCapture::SetCameraIntrinsics(float fx, float fy, float cx, float cy)
417 {
418  // If optical centers are not defined just use center of image
419  if (cx == -1)
420  {
421  this->cx = this->frame_width / 2.0f;
422  this->cy = this->frame_height / 2.0f;
423  }
424  else
425  {
426  this->cx = cx;
427  this->cy = cy;
428  }
429  // Use a rough guess-timate of focal length
430  if (fx == -1)
431  {
432  this->fx = 500.0f * (this->frame_width / 640.0f);
433  this->fy = 500.0f * (this->frame_height / 480.0f);
434 
435  this->fx = (this->fx + this->fy) / 2.0f;
436  this->fy = this->fx;
437  }
438  else
439  {
440  this->fx = fx;
441  this->fy = fy;
442  }

445 void SequenceCapture::CaptureThread()
446 {
447  int capacity = (CAPTURE_CAPACITY * 1024 * 1024) / (4 * frame_width * frame_height);
448  capture_queue.set_capacity(capacity);
449 
450  int frame_num_int = 0;
451 
452  while(capturing)
453  {
454  double timestamp_curr = 0;
455  cv::Mat tmp_frame;
456  cv::Mat_<uchar> tmp_gray_frame;
457 
458  if (!is_image_seq)
459  {
460  bool success = capture.read(tmp_frame);
461 
462  if (!success)
463  {
464  // Indicate lack of success by returning an empty image
465  tmp_frame = cv::Mat();
466  capturing = false;
467  }
468 
469  // Recording the timestamp
470  timestamp_curr = frame_num_int * (1.0 / fps);
471  }
472  else if (is_image_seq)
473  {
474  if (image_files.empty() || frame_num_int >= (int)image_files.size())
475  {
476  // Indicate lack of success by returning an empty image
477  tmp_frame = cv::Mat();
478  capturing = false;
479  }
480  else
481  {
482  tmp_frame = cv::imread(image_files[frame_num_int], cv::IMREAD_COLOR);
483  }
484  timestamp_curr = 0;
485  }
486 
487  frame_num_int++;
488  // Set the grayscale frame
489  ConvertToGrayscale_8bit(tmp_frame, tmp_gray_frame);
490 
491  capture_queue.push(std::make_tuple(timestamp_curr, tmp_frame, tmp_gray_frame));
492 
493  }
494 }
