116 // Contrast normalize the input for response map computation
117 void contrastNorm(const cv::Mat_<float>& input, cv::Mat_<float>& output)
118 {
119 
120  const unsigned int num_cols = input.cols;
121 
122  const unsigned int num_rows = input.rows;
123 
124  output = input.clone();
125 
126  cv::MatConstIterator_<float> p = input.begin();
127 
128  // Compute row wise
129  for (unsigned int y = 0; y < num_rows; ++y)
130  {
131 
132  cv::Scalar mean_s = cv::mean(input(cv::Rect(1,y,num_cols-1, 1)));
133  float mean = (float)mean_s[0];
134 
135  p++;
136 
137  float sum_sq = 0;
138  for (unsigned int x = 1; x < num_cols; ++x)
139  {
140  float curr = *p++;
141  sum_sq += (curr - mean) * (curr - mean);
142  }
143 
144  float norm = sqrt(sum_sq);
145 
146  if (norm == 0)
147  norm = 1;
148 
149  for (unsigned int x = 1; x < num_cols; ++x)
150  {
151  output.at<float>(y, x) = (output.at<float>(y, x) - mean) / norm;
152  }
153 
154  }

158 void im2colBias(const cv::Mat_<float>& input, const unsigned int width, const unsigned int height, cv::Mat_<float>& output)
159 {
160 
161  const unsigned int m = input.rows;
162  const unsigned int n = input.cols;
163 
164  // determine how many blocks there will be with a sliding window of width x height in the input
165  const unsigned int yB = m - height + 1;
166  const unsigned int xB = n - width + 1;
167 
168  // Allocate the output size
169  if(output.rows != xB*yB && output.cols != width * height + 1)
170  {
171  output = cv::Mat::ones(xB*yB, width * height + 1, CV_32F);
172  }
173 
174  // Iterate over the blocks
175  for (unsigned int j = 0; j< xB; j++)
176  {
177  for (unsigned int i = 0; i< yB; i++)
178  {
179  unsigned int rowIdx = i + j*yB;
180 
181  for (unsigned int yy = 0; yy < height; ++yy)
182  for (unsigned int xx = 0; xx < width; ++xx)
183  {
184  unsigned int colIdx = xx*height + yy;
185  output.at<float>(rowIdx, colIdx + 1) = input.at<float>(i + yy, j + xx);
186  }

192 void CEN_patch_expert::Response(const cv::Mat_<float> &area_of_interest, cv::Mat_<float> &response)
193 {
194 
195  int response_height = area_of_interest.rows - height_support + 1;
196  int response_width = area_of_interest.cols - width_support + 1;
197 
198  cv::Mat_<float> input_col;
199  im2colBias(area_of_interest, width_support, height_support, input_col);
200 
201  // Mean and standard deviation normalization
202  contrastNorm(input_col, response);
203  response = response.t();
204 
205  for (size_t layer = 0; layer < activation_function.size(); ++layer)
206  {
207 
208  // We are performing response = weights[layers] * response(t), but in OpenBLAS as that is significantly quicker than OpenCV
209  cv::Mat_<float> resp = response;
210  float* m1 = (float*)resp.data;
211  cv::Mat_<float> weight = weights[layer];
212  float* m2 = (float*)weight.data;
213 
214  cv::Mat_<float> resp_blas(weight.rows, resp.cols);
215  float* m3 = (float*)resp_blas.data;
216 
217  // Perform matrix multiplication in OpenBLAS (fortran call)
218  float alpha1 = 1.0;
219  float beta1 = 0.0;
220  char N[2]; N[0] = 'N';
221  sgemm_(N, N, &resp.cols, &weight.rows, &weight.cols, &alpha1, m1, &resp.cols, m2, &weight.cols, &beta1, m3, &resp.cols);
222 
223  // The above is a faster version of this, by calling the fortran version directly
224  //cblas_sgemm(CblasColMajor, CblasNoTrans, CblasNoTrans, resp.cols, weight.rows, weight.cols, 1, m1, resp.cols, m2, weight.cols, 0.0, m3, resp.cols);
225 
226  // Adding the bias (bit ugly, but the fastest way to do this)
227  response = resp_blas;
228 
229  float* data = (float*)response.data;
230  size_t height = response.rows;
231  size_t width = response.cols;
232  float* data_b = (float*)biases[layer].data;
233  for (size_t y = 0; y < height; ++y)
234  {
235  float bias = data_b[y];
236  for (size_t x = 0; x < width; ++x)
237  {
238  float in = *data + bias;
239  *data++ = in;
240  }
241  }
242 
243  // Perform activation and add bias at the same time
244  if (activation_function[layer] == 0) // Sigmoid
245  {
246 
247  size_t resp_size = response.rows * response.cols;
248 
249  // Iterate over the data directly
250  float* data = (float*)response.data;
251 
252  for (size_t counter = 0; counter < resp_size; ++counter)
253  {
254  float in = *data;
255  *data++ = 1.0 / (1.0 + exp(-(in)));
256  }
257 
258  }
259  else if (activation_function[layer] == 2)// ReLU
260  {
261  cv::threshold(response, response, 0, 0, cv::THRESH_TOZERO);
262  }
263 
264  }
265 
266  response = response.t();
267  response = response.reshape(1, response_height);
268  response = response.t();
269 
270 }

272 // Perform im2col, while at the same time doing contrast normalization and adding a bias term (also skip every other region)
273 void im2colBiasSparseContrastNorm(const cv::Mat_<float>& input, const unsigned int width, const unsigned int height, cv::Mat_<float>& output)
274 {
275  const unsigned int m = input.rows;
276  const unsigned int n = input.cols;
277 
278  // determine how many blocks there will be with a sliding window of width x height in the input
279  const unsigned int yB = m - height + 1;
280  const unsigned int xB = n - width + 1;
281 
282  // As we will be skipping half of the outputs
283  const unsigned int out_size = (yB*xB - 1) / 2;
284 
285  // Allocate the output size
286  if (output.rows != out_size && output.cols != width * height + 1)
287  {
288  output = cv::Mat::ones(out_size, width * height + 1, CV_32F);
289  }
290 
291  // Iterate over the blocks, skipping every second block
292  unsigned int rowIdx = 0;
293  unsigned int skipCounter = 0;
294  for (unsigned int j = 0; j< xB; j++)
295  {
296  for (unsigned int i = 0; i< yB; i++)
297  {
298  // Skip every second row
299  skipCounter++;
300  if ((skipCounter + 1) % 2 == 0)
301  {
302  continue;
303  }
304 
305  float* Mo = output.ptr<float>(rowIdx);
306 
307  float sum = 0;
308 
309  for (unsigned int yy = 0; yy < height; ++yy)
310  {
311  const float* Mi = input.ptr<float>(i + yy);
312  for (unsigned int xx = 0; xx < width; ++xx)
313  {
314  int colIdx = xx*height + yy;
315  float in = Mi[j + xx];
316  sum += in;
317 
318  Mo[colIdx+1] = in;
319  }
320  }
321 
322  // Working out the mean
323  float mean = sum / (float)(width * height);
324 
325  float sum_sq = 0;
326  const unsigned int num_items = width*height + 1;
327  // Working out the sum squared and subtracting the mean
328  for (unsigned int x = 1; x < num_items; ++x)
329  {
330  float in = Mo[x] - mean;
331  Mo[x] = in;
332  sum_sq += in * in;
333  }
334 
335  float norm = sqrt(sum_sq);
336 
337  // Avoiding division by 0
338  if (norm == 0)
339  {
340  norm = 1;
341  }
342 
343  // Flip multiplication to division for speed
344  norm = 1.0 / norm;
345 
346  for (unsigned int x = 1; x < num_items; ++x)
347  {
348  Mo[x] *= norm;
349  }
350 
351  rowIdx++;
352  }
353  }
354 }

402 // As the sparse patch expert output with interpolation, this function creates an interpolation matrix
403 void LandmarkDetector::interpolationMatrix(cv::Mat_<float>& mapMatrix, int response_height, int response_width, int input_width, int input_height)
404 {
405  int m = input_height;
406  int n = input_width;
407 
408  // determine how many blocks there will be with a sliding window of width x height in the input
409  int yB = m - 11 + 1;
410  int xB = n - 11 + 1;
411 
412  // As we will be skipping half of the outputs
413  int out_size = (yB*xB - 1) / 2;
414 
415  mapMatrix.create(out_size, response_height * response_width);
416  mapMatrix.setTo(0.0f);
417 
418  // Find a mapping from indices in the computed sparse response and the original full response
419  cv::Mat_<int> value_id_matrix(response_width, response_height, 0);
420 
421  int ind = 0;
422  for (int k = 0; k < value_id_matrix.rows * value_id_matrix.cols; ++k)
423  {
424  if (k % 2 != 0)
425  {
426  value_id_matrix.at<int>(k) = ind;
427  ind++;
428  }
429  }
430  value_id_matrix = value_id_matrix.t();
431 
432  int skip_counter = 0;
433  for (int x = 0; x < response_width; ++x)
434  {
435  for (int y = 0; y < response_height; ++y)
436  {
437  int mapping_col = x * response_height + y;
438  skip_counter++;
439  if (skip_counter % 2 == 0)
440  {
441  int val_id = value_id_matrix.at<int>(y, x);
442  mapMatrix.at<float>(val_id, mapping_col) = 1;
443  continue;
444  }
445 
446  float num_neigh = 0.0;
447  vector<int> val_ids;
448  if (x - 1 >= 0)
449  {
450  num_neigh++;
451  val_ids.push_back(value_id_matrix.at<int>(y, x - 1));
452  }
453  if (y - 1 >= 0)
454  {
455  num_neigh++;
456  val_ids.push_back(value_id_matrix.at<int>(y - 1, x));
457  }
458  if (x + 1 < response_width)
459  {
460  num_neigh++;
461  val_ids.push_back(value_id_matrix.at<int>(y, x + 1));
462  }
463  if (y + 1 < response_height)
464  {
465  num_neigh++;
466  val_ids.push_back(value_id_matrix.at<int>(y + 1, x));
467  }
468 
469  for (size_t k = 0; k < val_ids.size(); ++k)
470  {
471  mapMatrix.at<float>(val_ids[k], mapping_col) = 1.0 / num_neigh;
472  }
473  }
474  }
