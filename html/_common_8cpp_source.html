36 double Common::getSimilarity(const Mat &A, const Mat &B)
37 {
38  if (A.rows > 0 && A.rows == B.rows && A.cols > 0 && A.cols == B.cols)
39  {
40  // Calculate the L2 relative error between the 2 images.
41  double errorL2 = norm(A, B, NORM_L2);
42  // Convert to a reasonable scale, since L2 error is summed across all pixels of the image.
43  double similarity = errorL2 / (double)(A.rows * A.cols);
44  return similarity;
45  }
46  else
47  {
48  //cout << "WARNING: Images have a different size in 'getSimilarity()'." << endl;
49  return 100000000.0; // Return a bad value
