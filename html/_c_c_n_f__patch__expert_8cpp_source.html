81 // Compute sigmas for all landmarks for a particular view and window size
82 void CCNF_patch_expert::ComputeSigmas(std::vector<cv::Mat_<float> > sigma_components, int window_size)
83 {
84  for(size_t i=0; i < window_sizes.size(); ++i)
85  {
86  if( window_sizes[i] == window_size)
87  return;
88  }
89  // Each of the landmarks will have the same connections, hence constant number of sigma components
90  int n_betas = sigma_components.size();
91 
92  // calculate the sigmas based on alphas and betas
93  float sum_alphas = 0;
94 
95  int n_alphas = this->neurons.size();
96 
97  // sum the alphas first
98  for(int a = 0; a < n_alphas; ++a)
99  {
100  sum_alphas = sum_alphas + this->neurons[a].alpha;
101  }
102 
103  cv::Mat_<float> q1 = sum_alphas * cv::Mat_<float>::eye(window_size*window_size, window_size*window_size);
104 
105  cv::Mat_<float> q2 = cv::Mat_<float>::zeros(window_size*window_size, window_size*window_size);
106  for (int b=0; b < n_betas; ++b)
107  {
108  q2 = q2 + ((float)this->betas[b]) * sigma_components[b];
109  }
110 
111  cv::Mat_<float> SigmaInv = 2 * (q1 + q2);
112 
113  cv::Mat Sigma_f;
114  cv::invert(SigmaInv, Sigma_f, cv::DECOMP_CHOLESKY);
115 
116  window_sizes.push_back(window_size);
117  Sigmas.push_back(Sigma_f);
118 
119 }

138 // Perform im2col, while at the same time doing contrast normalization and adding a bias term
139 void im2colContrastNormBias(const cv::Mat_<float>& input, const unsigned int width, const unsigned int height, cv::Mat_<float>& output)
140 {
141  const unsigned int m = input.rows;
142  const unsigned int n = input.cols;
143 
144  // determine how many blocks there will be with a sliding window of width x height in the input
145  const unsigned int yB = m - height + 1;
146  const unsigned int xB = n - width + 1;
147 
148  // Allocate the output size
149  if (output.rows != xB*yB && output.cols != width * height + 1)
150  {
151  output = cv::Mat::ones(xB*yB, width * height + 1, CV_32F);
152  }
153 
154  // Iterate over the blocks
155  unsigned int rowIdx = 0;
156  for (unsigned int j = 0; j< xB; j++)
157  {
158  for (unsigned int i = 0; i< yB; i++)
159  {
160 
161  float* Mo = output.ptr<float>(rowIdx);
162 
163  float sum = 0;
164 
165  for (unsigned int yy = 0; yy < height; ++yy)
166  {
167  const float* Mi = input.ptr<float>(i + yy);
168  for (unsigned int xx = 0; xx < width; ++xx)
169  {
170  unsigned int colIdx = xx*height + yy;
171  float in = Mi[j + xx];
172  sum += in;
173 
174  Mo[colIdx + 1] = in;
175  }
176  }
177 
178  // Working out the mean
179  float mean = sum / (float)(width * height);
180 
181  float sum_sq = 0;
182  const unsigned int num_items = width*height + 1;
183  // Working out the sum squared and subtracting the mean
184  for (unsigned int x = 1; x < num_items; ++x)
185  {
186  float in = Mo[x] - mean;
187  Mo[x] = in;
188  sum_sq += in * in;
189  }
190 
191  float norm = sqrt(sum_sq);
192 
193  // Avoiding division by 0
194  if (norm == 0)
195  {
196  norm = 1;
197  }
198 
199  // Flip multiplication to division for speed
200  norm = 1.0 / norm;
201 
202  for (unsigned int x = 1; x < num_items; ++x)
203  {
204  Mo[x] *= norm;
205  }
206 
207  rowIdx++;
208  }
209  }
210 }

213 void CCNF_neuron::Response(const cv::Mat_<float> &im, cv::Mat_<double> &im_dft, cv::Mat &integral_img, cv::Mat &integral_img_sq, cv::Mat_<float> &resp)
214 {
215 
216  int h = im.rows - weights.rows + 1;
217  int w = im.cols - weights.cols + 1;
218 
219  // the patch area on which we will calculate reponses
220  cv::Mat_<float> I;
221 
222  if(neuron_type == 3)
223  {
224  // Perform normalisation across whole patch (ignoring the invalid values indicated by <= 0
225 
226  cv::Scalar mean;
227  cv::Scalar std;
228 
229  // ignore missing values
230  cv::Mat_<uchar> mask = im > 0;
231  cv::meanStdDev(im, mean, std, mask);
232 
233  // if all values the same don't divide by 0
234  if(std[0] != 0)
235  {
236  I = (im - mean[0]) / std[0];
237  }
238  else
239  {
240  I = (im - mean[0]);
241  }
242 
243  I.setTo(0, mask == 0);
244  }
245  else
246  {
247  if(neuron_type == 0)
248  {
249  I = im;
250  }
251  else
252  {
253  printf("ERROR(%s,%d): Unsupported patch type %d!\n", __FILE__,__LINE__,neuron_type);
254  abort();
255  }
256  }
257 
258  if(resp.empty())
259  {
260  resp.create(h, w);
261  }
262 
263  // The response from neuron before activation
264  if(neuron_type == 3)
265  {
266  // In case of depth we use per area, rather than per patch normalisation
267  matchTemplate_m(I, im_dft, integral_img, integral_img_sq, weights, weights_dfts, resp, cv::TM_CCOEFF); // the linear multiplication, efficient calc of response
268  }
269  else
270  {
271  matchTemplate_m(I, im_dft, integral_img, integral_img_sq, weights, weights_dfts, resp, cv::TM_CCOEFF_NORMED); // the linear multiplication, efficient calc of response
272  }
273 
274  cv::MatIterator_<float> p = resp.begin();
275 
276  cv::MatIterator_<float> q1 = resp.begin(); // respone for each pixel
277  cv::MatIterator_<float> q2 = resp.end();
278 
279  // the logistic function (sigmoid) applied to the response
280  while(q1 != q2)
281  {
282  *p++ = (2 * alpha) * 1.0 /(1.0 + exp( -(*q1++ * norm_weights + bias )));
283  }
284 
285 }

288 void CCNF_patch_expert::Read(ifstream &stream, std::vector<int> window_sizes, std::vector<std::vector<cv::Mat_<float> > > sigma_components)
289 {
290 
291  // Sanity check
292  int read_type;
293 
294  stream.read ((char*)&read_type, 4);
295  assert(read_type == 5);
296 
297  // the number of neurons for this patch
298  int num_neurons;
299  stream.read ((char*)&width, 4);
300  stream.read ((char*)&height, 4);
301  stream.read ((char*)&num_neurons, 4);
302 
303  if(num_neurons == 0)
304  {
305  // empty patch due to landmark being invisible at that orientation
306 
307  // read an empty int (due to the way things were written out)
308  stream.read ((char*)&num_neurons, 4);
309  return;
310  }
311 
312  neurons.resize(num_neurons);
313  for(int i = 0; i < num_neurons; i++)
314  neurons[i].Read(stream);
315 
316  // Combine the neuron weights to one weight matrix for more efficient computation
317  weight_matrix = cv::Mat_<float>(neurons.size(), 1 + neurons[0].weights.rows * neurons[0].weights.cols);
318  for (size_t i = 0; i < neurons.size(); i++)
319  {
320  cv::Mat_<float> w_tmp = neurons[i].weights.t();
321  cv::Mat_<float> weights_flat = w_tmp.reshape(1, neurons[i].weights.rows * neurons[i].weights.cols);
322  weights_flat = weights_flat.t();
323 
324  // Incorporate neuron weights directly
325  weights_flat = weights_flat * neurons[i].norm_weights;
326  weights_flat.copyTo(weight_matrix(cv::Rect(1, i, neurons[i].weights.rows * neurons[i].weights.cols, 1)));
327  // Incorporate bias as well
328  weight_matrix.at<float>(i, 0) = neurons[i].bias;
329  }
330 
331  // In case we are using OpenBLAS, make sure it is not multi-threading as we are multi-threading outside of it
332  openblas_set_num_threads(1);
333 
334  int n_sigmas = window_sizes.size();
335 
336  int n_betas = 0;
337 
338  if(n_sigmas > 0)
339  {
340  n_betas = sigma_components[0].size();
341 
342  betas.resize(n_betas);
343 
344  for (int i=0; i < n_betas; ++i)
345  {
346  stream.read ((char*)&betas[i], 8);
347  }
348  }
349 
350  // Read the patch confidence
351  stream.read ((char*)&patch_confidence, 8);

356 void CCNF_patch_expert::Response(const cv::Mat_<float> &area_of_interest, cv::Mat_<float> &response)
357 {
358 
359  int response_height = area_of_interest.rows - height + 1;
360  int response_width = area_of_interest.cols - width + 1;
361 
362  if(response.rows != response_height || response.cols != response_width)
363  {
364  response.create(response_height, response_width);
365  }
366 
367  response.setTo(0);
368 
369  // the placeholder for the DFT of the image, the integral image, and squared integral image so they don't get recalculated for every response
370  cv::Mat_<double> area_of_interest_dft;
371  cv::Mat integral_image, integral_image_sq;
372 
373  cv::Mat_<float> neuron_response;
374 
375  // responses from the neural layers
376  for(size_t i = 0; i < neurons.size(); i++)
377  {
378  // Do not bother with neuron response if the alpha is tiny and will not contribute much to overall result
379  if(neurons[i].alpha > 1e-4)
380  {
381 
382  neurons[i].Response(area_of_interest, area_of_interest_dft, integral_image, integral_image_sq, neuron_response);
383  response = response + neuron_response;
384  }
385  }
386 
387  int s_to_use = -1;
388 
389  // Find the matching sigma
390  for(size_t i=0; i < window_sizes.size(); ++i)
391  {
392  if(window_sizes[i] == response_height)
393  {
394  // Found the correct sigma
395  s_to_use = i;
396  break;
397  }
398  }
399 
400  cv::Mat_<float> resp_vec_f = response.reshape(1, response_height * response_width);
401 
402  cv::Mat out = Sigmas[s_to_use] * resp_vec_f;
403 
404  response = out.reshape(1, response_height);
405 
406  // Making sure the response does not have negative numbers
407  double min;
408 
409  minMaxIdx(response, &min, 0);
410  if(min < 0)
411  {
412  response = response - min;
413  }
414 
415 }
