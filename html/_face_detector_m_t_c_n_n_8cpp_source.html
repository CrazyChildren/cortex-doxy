139 {
140  if (input_img.channels() == 1)
141  {
142  cv::cvtColor(input_img, input_img, cv::COLOR_GRAY2BGR);
143  }
144 
145  int cnn_layer = 0;
146  int fully_connected_layer = 0;
147  int prelu_layer = 0;
148  int max_pool_layer = 0;
149 
150  // Slit a BGR image into three chnels
151  cv::Mat channels[3];
152  cv::split(input_img, channels);
153 
154  // Flip the BGR order to RGB
155  vector<cv::Mat_<float> > input_maps;
156  input_maps.push_back(channels[2]);
157  input_maps.push_back(channels[1]);
158  input_maps.push_back(channels[0]);

161 
162  for (size_t layer = 0; layer < cnn_layer_types.size(); ++layer)
163  {
164

166  int layer_type = cnn_layer_types[layer];
167 
168  // Convolutional layer
169  if (layer_type == 0)
170  {
171 
172  // Either perform direct convolution through matrix multiplication or use an FFT optimized version, which one is optimal depends on the kernel and input sizes
173  if (direct)
174  {
175  if(thread_safe)
176  {
177  cv::Mat_<float> pre_alloc;
178  convolution_direct_blas(outputs, input_maps, cnn_convolutional_layers_weights[cnn_layer], cnn_convolutional_layers[cnn_layer][0][0].rows, cnn_convolutional_layers[cnn_layer][0][0].cols, pre_alloc);
179  }
180  else
181  {

183  }
184 
185  }
186  else

194 
195  cnn_layer++;
196  }

203  int kernel_size_x = std::get<0>(cnn_max_pooling_layers[max_pool_layer]);

207  max_pool_layer++;
208  }
209  if (layer_type == 2)
210  {
211  fully_connected(outputs, input_maps, cnn_fully_connected_layers_weights[fully_connected_layer], cnn_fully_connected_layers_biases[fully_connected_layer]);
212  fully_connected_layer++;
213  }
214  if (layer_type == 3) // PReLU
215  {
216  // In place prelu computation
217  PReLU(input_maps, cnn_prelu_layer_weights[prelu_layer]);
218  outputs = input_maps;
219  prelu_layer++;
220  }
221  if (layer_type == 4)
222  {
223  outputs.clear();
224  for (size_t k = 0; k < input_maps.size(); ++k)
225  {
226  // Apply the sigmoid
227  cv::exp(-input_maps[k], input_maps[k]);
228  input_maps[k] = 1.0 / (1.0 + input_maps[k]);
229 
230  outputs.push_back(input_maps[k]);
231 
232  }
233  }
234  // Set the outputs of this layer to inputs of the next one
235  input_maps = outputs;
236  }
237 
238 
239  return outputs;

269 void CNN::Read(const string& location)
270 {
271 
272  openblas_set_num_threads(1);
273 
274  ifstream cnn_stream(location, ios::in | ios::binary);
275  if (cnn_stream.is_open())
276  {
277  cnn_stream.seekg(0, ios::beg);
278 
279  // Reading in CNNs
280 
281  int network_depth;
282  cnn_stream.read((char*)&network_depth, 4);
283 
284  cnn_layer_types.resize(network_depth);
285 
286  for (int layer = 0; layer < network_depth; ++layer)
287  {
288 
289  int layer_type;
290  cnn_stream.read((char*)&layer_type, 4);
291  cnn_layer_types[layer] = layer_type;
292 
293  // convolutional
294  if (layer_type == 0)
295  {
296 
297  // Read the number of input maps
298  int num_in_maps;
299  cnn_stream.read((char*)&num_in_maps, 4);
300 
301  // Read the number of kernels for each input map
302  int num_kernels;
303  cnn_stream.read((char*)&num_kernels, 4);
304 
305  vector<vector<cv::Mat_<float> > > kernels;
306 
307  kernels.resize(num_in_maps);
308 
309  vector<float> biases;
310  for (int k = 0; k < num_kernels; ++k)
311  {
312  float bias;
313  cnn_stream.read((char*)&bias, 4);
314  biases.push_back(bias);
315  }
316 
317  cnn_convolutional_layers_bias.push_back(biases);
318 
319  // For every input map
320  for (int in = 0; in < num_in_maps; ++in)
321  {
322  kernels[in].resize(num_kernels);
323 
324  // For every kernel on that input map
325  for (int k = 0; k < num_kernels; ++k)
326  {
327  ReadMatBin(cnn_stream, kernels[in][k]);
328 
329  }
330  }
331 
332  // Rearrange the kernels for faster inference with FFT
333  vector<vector<cv::Mat_<float> > > kernels_rearr;
334  kernels_rearr.resize(num_kernels);
335 
336  // Fill up the rearranged layer
337  for (int k = 0; k < num_kernels; ++k)
338  {
339  for (int in = 0; in < num_in_maps; ++in)
340  {
341  kernels_rearr[k].push_back(kernels[in][k]);
342  }
343  }
344 
345  cnn_convolutional_layers.push_back(kernels_rearr);
346 
347  // Place-holders for DFT precomputation
348  vector<map<int, vector<cv::Mat_<double> > > > cnn_convolutional_layers_dft_curr_layer;
349  cnn_convolutional_layers_dft_curr_layer.resize(num_kernels);
350  cnn_convolutional_layers_dft.push_back(cnn_convolutional_layers_dft_curr_layer);
351 
352  // Rearrange the flattened kernels into weight matrices for direct convolution computation
353  cv::Mat_<float> weight_matrix(num_in_maps * kernels_rearr[0][0].rows * kernels_rearr[0][0].cols, num_kernels);
354  for (int k = 0; k < num_kernels; ++k)
355  {
356  for (int i = 0; i < num_in_maps; ++i)
357  {
358  // Flatten the kernel
359  cv::Mat_<float> k_flat = kernels_rearr[k][i].t();
360  k_flat = k_flat.reshape(0, 1).t();
361  k_flat.copyTo(weight_matrix(cv::Rect(k, i * kernels_rearr[0][0].rows * kernels_rearr[0][0].cols, 1, kernels_rearr[0][0].rows * kernels_rearr[0][0].cols)));
362  }
363  }
364 
365  // Transpose the weight matrix for more convenient computation
366  weight_matrix = weight_matrix.t();
367 
368  // Add a bias term to the weight matrix for efficiency
369  cv::Mat_<float> W(weight_matrix.rows, weight_matrix.cols + 1, 1.0);
370  for (int k = 0; k < weight_matrix.rows; ++k)
371  {
372  W.at<float>(k, weight_matrix.cols) = biases[k];
373  }
374  weight_matrix.copyTo(W(cv::Rect(0, 0, weight_matrix.cols, weight_matrix.rows)));
375 
376  cnn_convolutional_layers_weights.push_back(W.t());
377  conv_layer_pre_alloc_im2col.push_back(cv::Mat_<float>());
378 
379  }
380  else if (layer_type == 1)
381  {
382  int kernel_x, kernel_y, stride_x, stride_y;
383  cnn_stream.read((char*)&kernel_x, 4);
384  cnn_stream.read((char*)&kernel_y, 4);
385  cnn_stream.read((char*)&stride_x, 4);
386  cnn_stream.read((char*)&stride_y, 4);
387  cnn_max_pooling_layers.push_back(std::tuple<int, int, int, int>(kernel_x, kernel_y, stride_x, stride_y));
388  }
389  else if (layer_type == 2)
390  {
391  cv::Mat_<float> biases;
392  ReadMatBin(cnn_stream, biases);
393  cnn_fully_connected_layers_biases.push_back(biases);
394 
395  // Fully connected layer
396  cv::Mat_<float> weights;
397  ReadMatBin(cnn_stream, weights);
398  cnn_fully_connected_layers_weights.push_back(weights.t());
399  }
400 
401  else if (layer_type == 3)
402  {
403  cv::Mat_<float> weights;
404  ReadMatBin(cnn_stream, weights);
405  cnn_prelu_layer_weights.push_back(weights);
406  }
407  }
408  }
409  else
410  {
411  cout << "WARNING: Can't find the CNN location" << endl;
412  }
413 }

474 // Perform non maximum supression on proposal bounding boxes prioritizing boxes with high score/confidence
475 std::vector<int> non_maximum_supression(const std::vector<cv::Rect_<float> >& original_bb, const std::vector<float>& scores, float thresh, bool minimum)
476 {
477 
478  // Sort the input bounding boxes by the detection score, using the nice trick of multimap always being sorted internally
479  std::multimap<float, size_t> idxs;
480  for (size_t i = 0; i < original_bb.size(); ++i)
481  {
482  idxs.insert(std::pair<float, size_t>(scores[i], i));
483  }
484 
485  std::vector<int> output_ids;
486 
487  // keep looping while some indexes still remain in the indexes list
488  while (idxs.size() > 0)
489  {
490  // grab the last rectangle
491  auto lastElem = --std::end(idxs);
492  size_t curr_id = lastElem->second;
493 
494  const cv::Rect& rect1 = original_bb[curr_id];
495 
496  idxs.erase(lastElem);
497 
498  // Iterate through remaining bounding boxes and choose which ones to remove
499  for (auto pos = std::begin(idxs); pos != std::end(idxs); )
500  {
501  // grab the current rectangle
502  const cv::Rect& rect2 = original_bb[pos->second];
503 
504  float intArea = (rect1 & rect2).area();
505  float unionArea;
506  if (minimum)
507  {
508  unionArea = cv::min(rect1.area(), rect2.area());
509  }
510  else
511  {
512  unionArea = rect1.area() + rect2.area() - intArea;
513  }
514  float overlap = intArea / unionArea;
515 
516  // Remove the bounding boxes with less confidence but with significant overlap with the current one
517  if (overlap > thresh)
518  {
519  pos = idxs.erase(pos);
520  }
521  else
522  {
523  ++pos;
524  }
525  }
526  output_ids.push_back(curr_id);
527 
528  }
529 
530  return output_ids;
531 
532 }

592 // Converting the bounding boxes to squares
593 void rectify(vector<cv::Rect_<float> >& total_bboxes)
594 {
595 
596  // Apply size and location offsets
597  for (size_t i = 0; i < total_bboxes.size(); ++i)
598  {
599  float height = total_bboxes[i].height;
600  float width = total_bboxes[i].width;
601 
602  float max_side = max(width, height);
603 
604  // Correct the starts based on new size
605  float new_min_x = total_bboxes[i].x + 0.5 * (width - max_side);
606  float new_min_y = total_bboxes[i].y + 0.5 * (height - max_side);
607 
608  total_bboxes[i].x = (int)new_min_x;
609  total_bboxes[i].y = (int)new_min_y;
610  total_bboxes[i].width = (int)max_side;
611  total_bboxes[i].height = (int)max_side;
612  }
