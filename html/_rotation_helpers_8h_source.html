46  // Using the XYZ convention R = Rx * Ry * Rz, left-handed positive sign
47  static cv::Matx33f Euler2RotationMatrix(const cv::Vec3f& eulerAngles)
48  {
49  cv::Matx33f rotation_matrix;
50 
51  float s1 = sin(eulerAngles[0]);
52  float s2 = sin(eulerAngles[1]);
53  float s3 = sin(eulerAngles[2]);
54 
55  float c1 = cos(eulerAngles[0]);
56  float c2 = cos(eulerAngles[1]);
57  float c3 = cos(eulerAngles[2]);
58 
59  rotation_matrix(0, 0) = c2 * c3;
60  rotation_matrix(0, 1) = -c2 *s3;
61  rotation_matrix(0, 2) = s2;
62  rotation_matrix(1, 0) = c1 * s3 + c3 * s1 * s2;
63  rotation_matrix(1, 1) = c1 * c3 - s1 * s2 * s3;
64  rotation_matrix(1, 2) = -c2 * s1;
65  rotation_matrix(2, 0) = s1 * s3 - c1 * c3 * s2;
66  rotation_matrix(2, 1) = c3 * s1 + c1 * s2 * s3;
67  rotation_matrix(2, 2) = c1 * c2;
68 
69  return rotation_matrix;
70  }

72  // Using the XYZ convention R = Rx * Ry * Rz, left-handed positive sign
73  static cv::Vec3f RotationMatrix2Euler(const cv::Matx33f& rotation_matrix)
74  {
75  float q0 = sqrt(1 + rotation_matrix(0, 0) + rotation_matrix(1, 1) + rotation_matrix(2, 2)) / 2.0f;
76  float q1 = (rotation_matrix(2, 1) - rotation_matrix(1, 2)) / (4.0f*q0);
77  float q2 = (rotation_matrix(0, 2) - rotation_matrix(2, 0)) / (4.0f*q0);
78  float q3 = (rotation_matrix(1, 0) - rotation_matrix(0, 1)) / (4.0f*q0);
79 
80  // Slower, but dealing with degenerate cases due to precision
81  float t1 = 2.0f * (q0*q2 + q1*q3);
82  if (t1 > 1) t1 = 1.0f;
83  if (t1 < -1) t1 = -1.0f;
84 
85  float yaw = asin(t1);
86  float pitch = atan2(2.0f * (q0*q1 - q2*q3), q0*q0 - q1*q1 - q2*q2 + q3*q3);
87  float roll = atan2(2.0f * (q0*q3 - q1*q2), q0*q0 + q1*q1 - q2*q2 - q3*q3);
88 
89  return cv::Vec3f(pitch, yaw, roll);
90  }

166  // Using Kabsch's algorithm for aligning shapes
167  //This assumes that align_from and align_to are already mean normalised
168  static cv::Matx22f AlignShapesKabsch2D(const cv::Mat_<float>& align_from, const cv::Mat_<float>& align_to)
169  {
170 
171  cv::SVD svd(align_from.t() * align_to);
172 
173  // make sure no reflection is there
174  // corr ensures that we do only rotaitons and not reflections
175  double d = cv::determinant(svd.vt.t() * svd.u.t());
176 
177  cv::Matx22f corr = cv::Matx22f::eye();
178  if (d > 0)
179  {
180  corr(1, 1) = 1;
181  }
182  else
183  {
184  corr(1, 1) = -1;
185  }
186 
187  cv::Matx22f R;
188  cv::Mat(svd.vt.t()*cv::Mat(corr)*svd.u.t()).copyTo(R);
189 
190  return R;
191  }
